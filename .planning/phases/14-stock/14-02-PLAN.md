    ```typescript
    // prisma/seed-wb-warehouses.ts
    // One-time seed: справочник WB складов с кластеризацией по 7 группам.
    // Список собран вручную через DevTools Network tab на seller.wildberries.ru (2026-04-21).
    // Cluster mapping согласован с пользователем в Zero Wave Plan 14-02.
    //
    // Запуск: npm run seed:wb-warehouses
    //
    // Идемпотентно: upsert by id. Повторный запуск обновляет name/cluster/shortCluster,
    // но не трогает needsClusterReview (сохраняется ручная пометка оператора).

    import { PrismaClient } from "@prisma/client"

    const prisma = new PrismaClient()

    interface WarehouseSeed {
      id: number
      name: string
      cluster: string      // Полное название из CLUSTER_FULL_NAMES
      shortCluster: string // Короткое: ЦФО / ЮГ / Урал / ПФО / СЗО / СФО / Прочие
    }

    // Список по согласованию с пользователем в Plan 14-02 Zero Wave:
    // [Вставить сюда hardcoded массив складов из Task 0 сообщения]
    const WB_WAREHOUSES: WarehouseSeed[] = [
      // ЦФО (Центральный федеральный округ) — Московская область, Тула, Тамбов, Калуга
      { id: 507, name: "Коледино", cluster: "Центральный федеральный округ", shortCluster: "ЦФО" },
      { id: 686, name: "Электросталь", cluster: "Центральный федеральный округ", shortCluster: "ЦФО" },
      // ... ВСЕ склады из согласованного списка Task 0

      // ЮГ (Южный + Северо-Кавказский ФО) — Краснодар, Ставрополь, Ростов
      { id: 117501, name: "Краснодар", cluster: "Южный ФО", shortCluster: "ЮГ" },

      // Урал (Уральский ФО) — Екатеринбург, Челябинск
      // ПФО (Приволжский ФО) — Казань, Самара, Уфа, Нижний Новгород
      // СЗО (Северо-Западный ФО) — Санкт-Петербург, Псков
      // СФО (Сибирский + Дальневосточный ФО) — Новосибирск, Хабаровск, Красноярск
      // Прочие — Калининград, зарубежные, неопознанные
    ]

    async function main() {
      console.log(`Начинаем seed ${WB_WAREHOUSES.length} WB складов...`)

      for (const w of WB_WAREHOUSES) {
        await prisma.wbWarehouse.upsert({
          where: { id: w.id },
          create: {
            id: w.id,
            name: w.name,
            cluster: w.cluster,
            shortCluster: w.shortCluster,
            isActive: true,
            needsClusterReview: false,
          },
          update: {
            name: w.name,
            cluster: w.cluster,
            shortCluster: w.shortCluster,
            // НЕ обновляем needsClusterReview — может быть ручная пометка оператора
          },
        })
      }

      // Summary по кластерам
      const counts: Record<string, number> = {}
      for (const w of WB_WAREHOUSES) {
        counts[w.shortCluster] = (counts[w.shortCluster] ?? 0) + 1
      }
      console.log("Seed по кластерам:")
      for (const [cluster, count] of Object.entries(counts)) {
        console.log(`  ${cluster}: ${count}`)
      }
      console.log(`Всего: ${WB_WAREHOUSES.length} складов`)
    }

    main()
      .catch((e) => {
        console.error(e)
        process.exit(1)
      })
      .finally(async () => {
        await prisma.$disconnect()
      })
    ```

    **КРИТИЧНО:** Массив `WB_WAREHOUSES` должен содержать ВСЕ склады из согласованного списка пользователя в Task 0 — не placeholder. Каждый склад с `shortCluster` из CLUSTER_ORDER (7 значений).

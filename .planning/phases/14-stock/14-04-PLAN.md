    ```typescript
    // lib/parse-ivanovo-excel.ts
    // Phase 14 (STOCK-11, STOCK-28): парсер Excel склада Иваново.
    // Колонки зафиксированы по реальной fixture в Plan 14-04 Zero Wave:
    //   [UPDATE_FROM_ZERO_WAVE] колонка {N1} = SKU, колонка {N2} = количество
    //
    // Паттерн: by-index (не by-name) — WB/внутренние Excel меняют заголовки между версиями.
    // Первая строка (index 0) = заголовки → пропускается.

    import * as XLSX from "xlsx"

    export interface ParsedIvanovoRow {
      sku: string       // сырой SKU до normalizeSku (для trace в error report)
      quantity: number
    }

    export interface ParseIvanovoResult {
      valid: ParsedIvanovoRow[]
      invalidRows: number[]  // номера строк с ошибкой (1-indexed для UX)
    }

    const COL_SKU = [УТОЧНИТЬ_ИЗ_ZERO_WAVE]   // индекс колонки SKU
    const COL_QTY = [УТОЧНИТЬ_ИЗ_ZERO_WAVE]   // индекс колонки количества

    export function parseIvanovoExcel(buf: Buffer): ParseIvanovoResult {
      const wb = XLSX.read(buf, { type: "buffer" })
      const sheetName = wb.SheetNames[0]
      if (!sheetName) return { valid: [], invalidRows: [] }

      const sheet = wb.Sheets[sheetName]!
      const rows = XLSX.utils.sheet_to_json<(string | number | null | undefined)[]>(
        sheet, { header: 1, defval: null }
      )

      const valid: ParsedIvanovoRow[] = []
      const invalidRows: number[] = []

      // Пропускаем строку 0 (заголовки)
      for (let i = 1; i < rows.length; i++) {
        const r = rows[i]
        if (!r || r.length === 0) continue

        const skuRaw = r[COL_SKU]
        const qtyRaw = r[COL_QTY]

        // Пустые строки (обе колонки null/empty) — просто skip, не invalid
        if ((skuRaw == null || skuRaw === "") && (qtyRaw == null || qtyRaw === "")) {
          continue
        }

        // Одна из колонок заполнена, другая пустая → invalid
        if (skuRaw == null || skuRaw === "" || qtyRaw == null || qtyRaw === "") {
          invalidRows.push(i + 1)  // 1-indexed для UX (Excel строки начинаются с 1)
          continue
        }

        const qty = parseInt(String(qtyRaw), 10)
        if (isNaN(qty) || qty < 0) {
          invalidRows.push(i + 1)
          continue
        }

        valid.push({
          sku: String(skuRaw).trim(),  // raw — normalizeSku в upsertIvanovoStock
          quantity: qty,
        })
      }

      return { valid, invalidRows }
    }
    ```

    ```typescript
    // app/api/stock/ivanovo-upload/route.ts
    // POST multipart → возвращает preview: {valid, unmatched, duplicates, invalidRows}
    // Фактическое применение — через upsertIvanovoStock server action из Dialog.

    import { NextRequest, NextResponse } from "next/server"
    import { requireSection } from "@/lib/rbac"
    import { prisma } from "@/lib/prisma"
    import { parseIvanovoExcel } from "@/lib/parse-ivanovo-excel"
    import { normalizeSku } from "@/lib/normalize-sku"

    export const runtime = "nodejs"  // XLSX не работает в Edge

    export async function POST(req: NextRequest) {
      await requireSection("STOCK", "MANAGE")

      const formData = await req.formData()
      const file = formData.get("file")
      if (!(file instanceof File)) {
        return NextResponse.json({ error: "Файл не загружен" }, { status: 400 })
      }

      const buf = Buffer.from(await file.arrayBuffer())
      let parsed
      try {
        parsed = parseIvanovoExcel(buf)
      } catch (e) {
        return NextResponse.json({
          error: `Не удалось прочитать Excel: ${(e as Error).message}. Проверьте формат.`
        }, { status: 400 })
      }

      // Normalize SKUs + dedup detection + lookup in DB
      const normalizedRows: Array<{ skuRaw: string; sku: string | null; quantity: number; normalizeError?: string }> = []
      for (const row of parsed.valid) {
        try {
          const sku = normalizeSku(row.sku)
          normalizedRows.push({ skuRaw: row.sku, sku, quantity: row.quantity })
        } catch (e) {
          normalizedRows.push({ skuRaw: row.sku, sku: null, quantity: row.quantity, normalizeError: (e as Error).message })
        }
      }

      // Detect duplicates (same sku appears N>1)
      const skuCounts = new Map<string, number>()
      for (const r of normalizedRows) {
        if (r.sku) skuCounts.set(r.sku, (skuCounts.get(r.sku) ?? 0) + 1)
      }
      const duplicateSkus = new Set([...skuCounts.entries()].filter(([_, c]) => c > 1).map(([s]) => s))

      // Lookup existing Products by sku
      const uniqueSkus = [...new Set(normalizedRows.filter(r => r.sku).map(r => r.sku!))]
      const existingProducts = await prisma.product.findMany({
        where: { sku: { in: uniqueSkus }, deletedAt: null },
        select: { id: true, sku: true, name: true, ivanovoStock: true },
      })
      const productsBySku = new Map(existingProducts.map(p => [p.sku, p]))

      // Build preview response
      const valid: Array<{ sku: string; productId: string; productName: string; oldQty: number | null; newQty: number }> = []
      const unmatched: Array<{ skuRaw: string; sku: string; newQty: number }> = []
      const invalid: Array<{ skuRaw: string; error: string }> = []
      const duplicates: Array<{ sku: string; occurrences: number; lastQty: number }> = []

      for (const r of normalizedRows) {
        if (!r.sku) {
          invalid.push({ skuRaw: r.skuRaw, error: r.normalizeError ?? "unknown" })
          continue
        }
        const product = productsBySku.get(r.sku)
        if (!product) {
          unmatched.push({ skuRaw: r.skuRaw, sku: r.sku, newQty: r.quantity })
          continue
        }
        // Matched — if duplicate, последнее значение выигрывает (last-write-wins)
        valid.push({
          sku: r.sku,
          productId: product.id,
          productName: product.name,
          oldQty: product.ivanovoStock,
          newQty: r.quantity,
        })
      }

      // Duplicates — по sku, только те что matched
      for (const sku of duplicateSkus) {
        const product = productsBySku.get(sku)
        const last = normalizedRows.filter(r => r.sku === sku).at(-1)!
        duplicates.push({
          sku,
          occurrences: skuCounts.get(sku)!,
          lastQty: last.quantity,
        })
      }

      return NextResponse.json({
        valid,
        unmatched,
        duplicates,
        invalid,
        invalidParseRows: parsed.invalidRows,
      })
    }
    ```

    ```typescript
    // app/actions/stock.ts
    // Phase 14 server actions для /stock: upsertIvanovoStock, updateProductionStock (14-05), updateTurnoverNorm (14-05)

    "use server"

    import { prisma } from "@/lib/prisma"
    import { requireSection } from "@/lib/rbac"
    import { revalidatePath } from "next/cache"
    import { z } from "zod"

    const IvanovoRowSchema = z.object({
      sku: z.string().min(1),
      quantity: z.number().int().min(0).max(999999),
    })

    export type UpsertIvanovoResult = {
      imported: number
      notFound: string[]  // SKU которых нет в БД
      errors: Array<{ sku: string; error: string }>
    }

    export async function upsertIvanovoStock(
      rows: Array<{ sku: string; quantity: number }>
    ): Promise<UpsertIvanovoResult> {
      await requireSection("STOCK", "MANAGE")

      const parsedRows = rows.map(r => IvanovoRowSchema.safeParse(r))
      const valid = parsedRows.filter(r => r.success).map(r => r.data as z.infer<typeof IvanovoRowSchema>)

      const result: UpsertIvanovoResult = {
        imported: 0,
        notFound: [],
        errors: [],
      }

      const now = new Date()
      await prisma.$transaction(async (tx) => {
        for (const row of valid) {
          try {
            const updated = await tx.product.updateMany({
              where: { sku: row.sku, deletedAt: null },
              data: {
                ivanovoStock: row.quantity,
                ivanovoStockUpdatedAt: now,
              },
            })
            if (updated.count === 0) {
              result.notFound.push(row.sku)
            } else {
              result.imported += updated.count
            }
          } catch (e) {
            result.errors.push({ sku: row.sku, error: (e as Error).message })
          }
        }
      })

      revalidatePath("/stock")
      return result
    }
    ```

    ```typescript
    import { useState, useRef } from "react"

    export function IvanovoUploadButton() {

      const [preview, setPreview] = useState<PreviewData | null>(null)
      const fileInput = useRef<HTMLInputElement>(null)

      const handleFile = async (file: File) => {
        setIsParsing(true)
        try {
          const fd = new FormData()
          fd.append("file", file)
          const res = await fetch("/api/stock/ivanovo-upload", { method: "POST", body: fd })
          if (!res.ok) {
            const { error } = await res.json().catch(() => ({ error: `HTTP ${res.status}` }))

            return
          }
          const data: PreviewData = await res.json()
          setPreview(data)
        } catch (e) {
          toast.error(`Не удалось загрузить файл: ${(e as Error).message}`)
        } finally {
          setIsParsing(false)
          if (fileInput.current) fileInput.current.value = ""
        }
      }

      return (
        <>
          <input
            ref={fileInput}
            type="file"
            accept=".xlsx"
            className="hidden"
            onChange={(e) => {
              const f = e.target.files?.[0]
              if (f) handleFile(f)
            }}
          />
          <Button
            variant="outline"
            size="sm"
            disabled={isParsing}
            onClick={() => fileInput.current?.click()}
          >
            <Upload className="mr-2 h-4 w-4" />
            {isParsing ? "Загрузка..." : "Загрузить Excel Иваново"}
          </Button>

              onClose={() => setPreview(null)}
    ```

    ```typescript
    "use client"

    import { useState, useTransition } from "react"
    import { useRouter } from "next/navigation"
    import { toast } from "sonner"
    import {
      Dialog,
      DialogContent,
      DialogHeader,
      DialogTitle,
      DialogDescription,
      DialogFooter,
    } from "@/components/ui/dialog"
    import { Button } from "@/components/ui/button"
    import { upsertIvanovoStock } from "@/app/actions/stock"

    export interface PreviewData {
      valid: Array<{ sku: string; productId: string; productName: string; oldQty: number | null; newQty: number }>
      unmatched: Array<{ skuRaw: string; sku: string; newQty: number }>
      duplicates: Array<{ sku: string; occurrences: number; lastQty: number }>
      invalid: Array<{ skuRaw: string; error: string }>
      invalidParseRows: number[]
    }

    export function IvanovoUploadDialog({ preview, onClose }: {
      preview: PreviewData
      onClose: () => void
    }) {
      const [isApplying, startTransition] = useTransition()
      const router = useRouter()
      const validCount = preview.valid.length

      const onApply = () => {
        startTransition(async () => {
          try {
            const rows = preview.valid.map(v => ({ sku: v.sku, quantity: v.newQty }))
            const result = await upsertIvanovoStock(rows)
            toast.success(`Импортировано ${result.imported} строк остатков Иваново`)
            if (result.notFound.length > 0) {
              toast.warning(`${result.notFound.length} SKU не найдены в БД`)
            }
            onClose()
            router.refresh()
          } catch (e) {
            toast.error(`Ошибка применения: ${(e as Error).message}`)
          }
        })
      }

      return (
        <Dialog open onOpenChange={(o) => !o && onClose()}>
          <DialogContent className="sm:max-w-2xl max-h-[90vh] overflow-y-auto">
            <DialogHeader>
              <DialogTitle>Импорт остатков склада Иваново</DialogTitle>
              <DialogDescription>
                Предварительный просмотр изменений перед применением
              </DialogDescription>
            </DialogHeader>

            {/* Секция 1 — Корректные строки с diff */}
            <div className="space-y-2">
              <h3 className="text-sm font-medium">Изменения ({validCount})</h3>
              <div className="rounded border divide-y max-h-60 overflow-y-auto">
                {preview.valid.map((row) => (
                  <div key={row.sku} className="flex items-center justify-between px-3 py-2 text-sm">
                    <span>{row.sku} — {row.productName}</span>
                    <span className="tabular-nums">
                      <span className="text-muted-foreground">{row.oldQty ?? "—"}</span>
                      {" → "}
                      <span className="font-medium">{row.newQty}</span>
                    </span>
                  </div>
                ))}
              </div>
            </div>

            {/* Секция 2 — Unmatched */}
            {preview.unmatched.length > 0 && (
              <div className="space-y-2">
                <h3 className="text-sm font-medium text-yellow-600">Не найдено в базе ({preview.unmatched.length})</h3>
                <p className="text-xs text-muted-foreground">Эти строки будут пропущены, но не заблокируют импорт.</p>
                <div className="rounded border bg-muted/30 max-h-32 overflow-y-auto text-xs">
                  {preview.unmatched.map((u, i) => (
                    <div key={i} className="px-3 py-1">{u.sku} (из "{u.skuRaw}") — qty {u.newQty}</div>
                  ))}
                </div>
              </div>
            )}

            {/* Секция 3 — Дубликаты */}
            {preview.duplicates.length > 0 && (
              <div className="space-y-2">
                <h3 className="text-sm font-medium text-yellow-600">Дубликаты в файле ({preview.duplicates.length})</h3>
                <p className="text-xs text-muted-foreground">Применится последнее значение каждого SKU.</p>
                <div className="rounded border bg-muted/30 max-h-32 overflow-y-auto text-xs">
                  {preview.duplicates.map((d) => (
                    <div key={d.sku} className="px-3 py-1">{d.sku} — {d.occurrences}× (применится qty {d.lastQty})</div>
                  ))}
                </div>
              </div>
            )}

            {/* Секция 4 — Невалидные */}
            {(preview.invalid.length > 0 || preview.invalidParseRows.length > 0) && (
              <div className="space-y-2">
                <h3 className="text-sm font-medium text-destructive">Невалидные строки ({preview.invalid.length + preview.invalidParseRows.length})</h3>
                <div className="rounded border bg-muted/30 max-h-32 overflow-y-auto text-xs">
                  {preview.invalid.map((inv, i) => (
                    <div key={i} className="px-3 py-1">{inv.skuRaw} — {inv.error}</div>
                  ))}
                  {preview.invalidParseRows.map((r) => (
                    <div key={r} className="px-3 py-1">Строка {r} — пустое SKU или некорректное количество</div>
                  ))}
                </div>
              </div>
            )}

            <DialogFooter>
              <Button variant="outline" onClick={onClose} disabled={isApplying}>Отмена</Button>
              <Button
                onClick={onApply}
                disabled={validCount === 0 || isApplying}
                title={validCount === 0 ? "Нет корректных строк для импорта" : undefined}
              >
                {isApplying ? "Применить..." : `Применить (${validCount} строк)`}
              </Button>
            </DialogFooter>
          </DialogContent>
        </Dialog>
      )
    }
    ```

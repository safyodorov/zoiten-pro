    **createTemplate(input)**:
    - Проверяет requireSection("SUPPORT", "MANAGE") → если не SUPERADMIN/MANAGE → throw
    - Zod: name[2..80], text[1..5000], channel in [FEEDBACK, QUESTION, CHAT], situationTag? max 60, nmId? positive int, isActive default true
    - При channel=RETURN или MESSENGER → ZodError «Канал должен быть FEEDBACK, QUESTION или CHAT»
    - prisma.responseTemplate.create с createdById/updatedById = current userId (null если session.user.id отсутствует — допустимо, FK onDelete: SetNull)
    - revalidatePath("/support/templates")
    - Возвращает { ok: true, id }

    **updateTemplate(id, input)**:
    - Те же guards + Zod
    - prisma.responseTemplate.update({ where: { id }, data: { ...validated, updatedById: userId } }) — updatedAt авто
    - Ловит Prisma P2025 (not found) и P2002 (unique violation) с понятными сообщениями
    - revalidatePath

    **deleteTemplate(id)**:
    - requireSection + hard delete prisma.responseTemplate.delete({ where: { id } })
    - При P2025 → «Шаблон не найден»
    - revalidatePath

    **toggleTemplateActive(id)**:
    - requireSection + read current isActive → update to !isActive
    - Возвращает новый isActive в result

    **exportTemplatesJson()**:
    - requireSection (read достаточно — но мы хотим SUPPORT+MANAGE чтобы случайно не выгружал VIEWER)
    - Читает все шаблоны (isActive=true)
    - Возвращает JSON-строку: `{ "version": 1, "exportedAt": ISO_DATE, "templates": [{ name, text, channel, situationTag, nmId, isActive }] }` — без id/createdById/timestamps (переносимость)

    **importTemplatesJson(json)**:
    - requireSection
    - JSON.parse + Zod валидация: { version: 1, templates: Array<TemplateImportItem> }
    - Для каждого template: upsert по where @@unique([name, channel]) — если существует → update, иначе → create с createdById=userId
    - Возвращает { ok: true, added: N, updated: M, errors: [{name, channel, reason}] } — errors[] содержит Zod-ошибки конкретных записей (не роняет транзакцию)

    **Шаг A — создать `app/actions/templates.ts`:**

    ```typescript
    "use server"

    // Server actions CRUD шаблонов ответов + Export/Import JSON.
    // WB /api/v1/templates отключён 2025-11-19 — все операции локальные.
    // Export/Import JSON заменяет WB sync (per user decision D-01).

    import { revalidatePath } from "next/cache"
    import { prisma } from "@/lib/prisma"
    import { requireSection } from "@/lib/rbac"
    import { auth } from "@/lib/auth"
    import { z } from "zod"
    import { Prisma } from "@prisma/client"

    export type ActionResult<T = void> =
      | ({ ok: true } & (T extends void ? {} : T))
      | { ok: false; error: string }

    const templateSchema = z.object({
      name: z.string().trim().min(2, "Имя короче 2 символов").max(80, "Имя длиннее 80 символов"),
      text: z.string().trim().min(1, "Текст пустой").max(5000, "Текст длиннее 5000 символов"),
      channel: z.enum(["FEEDBACK", "QUESTION", "CHAT"] as const, {
        errorMap: () => ({ message: "Канал должен быть FEEDBACK, QUESTION или CHAT" }),
      }),
      situationTag: z.string().trim().max(60).nullable().optional(),
      nmId: z.coerce.number().int().positive().nullable().optional(),
      isActive: z.boolean().default(true),
    })

    async function getSessionUserId(): Promise<string | null> {
      const session = await auth()
      return session?.user?.id ?? null
    }

    export async function createTemplate(
      input: z.input<typeof templateSchema>
    ): Promise<ActionResult<{ id: string }>> {
      try {
        await requireSection("SUPPORT", "MANAGE")
        const userId = await getSessionUserId()
        const data = templateSchema.parse(input)
        const created = await prisma.responseTemplate.create({
          data: { ...data, createdById: userId, updatedById: userId },
        })
        revalidatePath("/support/templates")
        return { ok: true, id: created.id }
      } catch (err) {
        if (err instanceof Prisma.PrismaClientKnownRequestError && err.code === "P2002") {
          return { ok: false, error: "Шаблон с таким именем уже существует в этом канале" }
        }
        return { ok: false, error: err instanceof Error ? err.message : "Ошибка" }
      }
    }

    export async function updateTemplate(
      id: string,
      input: z.input<typeof templateSchema>
    ): Promise<ActionResult> {
      try {
        await requireSection("SUPPORT", "MANAGE")
        const userId = await getSessionUserId()
        const data = templateSchema.parse(input)
        await prisma.responseTemplate.update({
          where: { id },
          data: { ...data, updatedById: userId },
        })
        revalidatePath("/support/templates")
        return { ok: true }
      } catch (err) {
        if (err instanceof Prisma.PrismaClientKnownRequestError) {
          if (err.code === "P2025") return { ok: false, error: "Шаблон не найден" }
          if (err.code === "P2002") return { ok: false, error: "Имя шаблона уже используется в этом канале" }
        }
        return { ok: false, error: err instanceof Error ? err.message : "Ошибка" }
      }
    }

    export async function deleteTemplate(id: string): Promise<ActionResult> {
      try {
        await requireSection("SUPPORT", "MANAGE")
        await prisma.responseTemplate.delete({ where: { id } })
        revalidatePath("/support/templates")
        return { ok: true }
      } catch (err) {
        if (err instanceof Prisma.PrismaClientKnownRequestError && err.code === "P2025") {
          return { ok: false, error: "Шаблон не найден" }
        }
        return { ok: false, error: err instanceof Error ? err.message : "Ошибка" }
      }
    }

    export async function toggleTemplateActive(
      id: string
    ): Promise<ActionResult<{ isActive: boolean }>> {
      try {
        await requireSection("SUPPORT", "MANAGE")
        const current = await prisma.responseTemplate.findUnique({
          where: { id },
          select: { isActive: true },
        })
        if (!current) return { ok: false, error: "Шаблон не найден" }
        const updated = await prisma.responseTemplate.update({
          where: { id },
          data: { isActive: !current.isActive },
          select: { isActive: true },
        })
        revalidatePath("/support/templates")
        return { ok: true, isActive: updated.isActive }
      } catch (err) {
        return { ok: false, error: err instanceof Error ? err.message : "Ошибка" }
      }
    }

    // ── Export / Import JSON ─────────────────────────────────────────
    // Заменяет WB sync (D-01) — WB Templates API отключён 2025-11-19.

    const EXPORT_VERSION = 1

    export async function exportTemplatesJson(): Promise<ActionResult<{ json: string }>> {
      try {
        await requireSection("SUPPORT", "MANAGE")
        const templates = await prisma.responseTemplate.findMany({
          where: { isActive: true },
          select: { name: true, text: true, channel: true, situationTag: true, nmId: true, isActive: true },
          orderBy: [{ channel: "asc" }, { name: "asc" }],
        })
        const payload = {
          version: EXPORT_VERSION,
          exportedAt: new Date().toISOString(),
          templates,
        }
        return { ok: true, json: JSON.stringify(payload, null, 2) }
      } catch (err) {
        return { ok: false, error: err instanceof Error ? err.message : "Ошибка" }
      }
    }

    const importItemSchema = templateSchema.extend({ isActive: z.boolean().optional().default(true) })
    const importSchema = z.object({
      version: z.literal(EXPORT_VERSION),
      templates: z.array(z.unknown()),  // Валидируем каждый item отдельно для сбора errors
    })

    export async function importTemplatesJson(
      json: string
    ): Promise<ActionResult<{ added: number; updated: number; errors: Array<{ name?: string; channel?: string; reason: string }> }>> {
      try {
        await requireSection("SUPPORT", "MANAGE")
        const userId = await getSessionUserId()

        let parsed: unknown
        try {
          parsed = JSON.parse(json)
        } catch {
          return { ok: false, error: "Невалидный JSON" }
        }

        const envelope = importSchema.safeParse(parsed)
        if (!envelope.success) {
          return { ok: false, error: `Неподдерживаемый формат: ${envelope.error.message.slice(0, 200)}` }
        }

        let added = 0
        let updated = 0
        const errors: Array<{ name?: string; channel?: string; reason: string }> = []

        for (const raw of envelope.data.templates) {
          const itemResult = importItemSchema.safeParse(raw)
          if (!itemResult.success) {
            const rawName = (raw as { name?: unknown } | null)?.name
            errors.push({
              name: typeof rawName === "string" ? rawName : undefined,
              reason: itemResult.error.issues[0]?.message ?? "Невалидная запись",
            })
            continue
          }
          const data = itemResult.data
          try {
            const result = await prisma.responseTemplate.upsert({
              where: { name_channel: { name: data.name, channel: data.channel } },
              create: { ...data, createdById: userId, updatedById: userId },
              update: {
                text: data.text,
                situationTag: data.situationTag,
                nmId: data.nmId,
                isActive: data.isActive,
                updatedById: userId,
              },
              select: { createdAt: true, updatedAt: true },
            })
            // upsert не говорит, был ли insert или update — сравниваем timestamps
            if (result.createdAt.getTime() === result.updatedAt.getTime()) added++
            else updated++
          } catch (err) {
            errors.push({
              name: data.name,
              channel: data.channel,
              reason: err instanceof Error ? err.message : "DB ошибка",
            })
          }
        }

        revalidatePath("/support/templates")
        return { ok: true, added, updated, errors }
      } catch (err) {
        return { ok: false, error: err instanceof Error ? err.message : "Ошибка" }
      }
    }
    ```

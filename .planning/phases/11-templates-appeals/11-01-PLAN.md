    // Phase 11 — Шаблоны ответов (local-only, WB API отключён 2025-11-19)
    // ──────────────────────────────────────────────────────────────────
    // Хранилище 100% локальное — WB endpoints /api/v1/templates отключены.

    // Канал ограничен Zod'ом в server action до FEEDBACK/QUESTION/CHAT (не RETURN/MESSENGER).
    model ResponseTemplate {
      id            String        @id @default(cuid())
      name          String        // уникальное имя в рамках канала
      text          String        @db.Text
      channel       TicketChannel // FEEDBACK | QUESTION | CHAT (валидация в Zod)
      situationTag  String?       // свободный тег «Положительный», «Негативный 1★», «Размер»
      nmId          Int?          // опциональная привязка к товару (без FK — паттерн проекта)
      isActive      Boolean       @default(true)
      createdById   String?
      createdBy     User?         @relation("TemplateCreator", fields: [createdById], references: [id], onDelete: SetNull)
      updatedById   String?
      updatedBy     User?         @relation("TemplateUpdater", fields: [updatedById], references: [id], onDelete: SetNull)
      createdAt     DateTime      @default(now())
      updatedAt     DateTime      @updatedAt

      @@unique([name, channel])
      @@index([channel, isActive])
      @@index([nmId])
      @@index([situationTag])
    }
    ```

    Rationale: `@@unique([name, channel])` позволяет использовать одно имя в разных каналах (например «Спасибо» в FEEDBACK и CHAT) + это ключ для import JSON upsert (Plan 11-02).

    // ──────────────────────────────────────────────────────────────────
    // Phase 11 — Трекер обжалований (hybrid manual, WB API отключён 2025-12-08)
    // ──────────────────────────────────────────────────────────────────
    // WB endpoint POST /api/v1/feedbacks/actions отключён — менеджер создаёт
    // локальную запись + вручную подаёт жалобу в ЛК WB через jump-link.
    // Ручной переключатель статуса (PENDING → APPROVED|REJECTED) после ответа WB.
    // Один ticket → один AppealRecord (повтор через API невозможен, локальный повтор = update).
    model AppealRecord {
      id               String        @id @default(cuid())
      ticketId         String        @unique
      ticket           SupportTicket @relation("Appeal", fields: [ticketId], references: [id], onDelete: Cascade)
      reason           String        // label причины из APPEAL_REASONS (денормализованно — чтобы изменение справочника не ломало историю)
      text             String        @db.Text // свободный комментарий менеджера 10-1000 символов
      status           AppealStatus  @default(PENDING)
      createdById      String
      createdBy        User          @relation("AppealCreator", fields: [createdById], references: [id], onDelete: Restrict)
      resolvedById     String?
      resolvedBy       User?         @relation("AppealResolver", fields: [resolvedById], references: [id], onDelete: SetNull)
      createdAt        DateTime      @default(now())
      updatedAt        DateTime      @updatedAt
      appealResolvedAt DateTime?

      @@index([status])
      @@index([createdAt])
    }
    ```

    Rationale:
    - `@@unique([ticketId])` — 1:1 с тикетом (повторная жалоба через API невозможна).

    - `resolvedBy onDelete: SetNull` — если менеджер деактивирован, запись остаётся.
    - `reason: String` (денормализованный label, не id) — устойчиво к изменениям справочника в коде.

    **Шаг B — создать `lib/template-vars.ts`:**

    ```typescript
    // Подстановка переменных шаблона {имя_покупателя} и {название_товара}.
    // Чистая функция, вызывается на клиенте в TemplatePickerModal.onPick перед записью в textarea.
    // Консистентна с Phase 10 AutoReplyConfig.messageText (оба используют те же placeholder'ы).
    // Fallback: customerName → «покупатель», productName → «» (пусто).

    export interface TemplateVarContext {
      customerName?: string | null
      productName?: string | null
    }

    export function substituteTemplateVars(
      text: string,
      ctx: TemplateVarContext
    ): string {
      const customer = ctx.customerName?.trim() || "покупатель"
      const product = ctx.productName?.trim() || ""
      return text
        .replace(/\{имя_покупателя\}/g, customer)
        .replace(/\{название_товара\}/g, product)
    }
    ```

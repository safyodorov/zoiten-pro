```typescript
export function parseDDMMYYYY(s: string | number | null): Date | null {
  if (s == null) return null
  const m = String(s).trim().match(/^(\d{1,2})\.(\d{1,2})\.(\d{4})$/)
  if (!m) return null
  return new Date(Date.UTC(parseInt(m[3]), parseInt(m[2]) - 1, parseInt(m[1])))
}

export function parseAmount(v: string | number | null): number | null {
  if (v == null || v === "") return null
  if (typeof v === "number") return Number.isFinite(v) ? v : null
  // убрать разделители тысяч (запятые перед группами из 3 цифр), точка = десятичная
  const cleaned = String(v).replace(/\s/g, "").replace(/,(?=\d{3}(\.|,|$))/g, "")
  const n = parseFloat(cleaned)
  return Number.isFinite(n) ? n : null
}

export function normalizePurpose(s: string | null | undefined): string {
  return String(s ?? "").trim().replace(/\s+/g, " ").toLowerCase()
}

export function extractBic(s: string | null | undefined): string | null {
  if (!s) return null
  const m = String(s).match(/БИК\s*(\d{9})/i) ?? String(s).match(/\b(\d{9})\b/)
  return m ? m[1] : null
}

export function buildHeaderMap(headerRow: (string | number | null)[]): Record<string, number> {
  const map: Record<string, number> = {}
  headerRow.forEach((cell, idx) => {
    if (cell != null && String(cell).trim() !== "") map[String(cell).trim()] = idx
  })
  return map
}
```

**`lib/bank-import/fingerprint.ts`:**

```typescript
import { createHash } from "crypto"
import { normalizePurpose } from "./normalize"
import type { ParsedTransaction } from "./types"

export function buildFingerprintFields(tx: ParsedTransaction): string[] {
  return [
    tx.accountNumber,
    tx.date.toISOString().slice(0, 10),   // YYYY-MM-DD
    tx.direction,
    tx.amount.toFixed(2),
    tx.docNumber ?? "",
    tx.counterpartyInn ?? "",
    normalizePurpose(tx.purpose),
  ]
}

export function computeFingerprint(tx: ParsedTransaction): string {
  return createHash("sha256").update(buildFingerprintFields(tx).join("|")).digest("hex")
}
```

```typescript
import * as XLSX from "xlsx"
import { parseVtbStatement } from "./vtb-adapter"
import { parsePsbStatement } from "./psb-adapter"
import { parseSberStatement } from "./sber-adapter"
import type { BankFormat, ParsedTransaction, ParseResult } from "./types"

export * from "./types"
export { computeFingerprint, buildFingerprintFields } from "./fingerprint"
export { parseVtbStatement, parsePsbStatement, parseSberStatement }

export function detectFormat(fileName: string, workbook: XLSX.WorkBook): BankFormat {
  // 1) по имени файла (надёжнее всего для ВТБ)
  if (/^VTB_BankStatement/i.test(fileName)) return "vtb"
  if (/^СберБизнес/i.test(fileName) || /Сбербизнес/i.test(fileName)) return "sber"
  if (/Выписка по счету/i.test(fileName)) return "psb"
  // 2) fallback по сигнатуре шапки первого листа
  const sheet = workbook.Sheets[workbook.SheetNames[0]]
  const rows = XLSX.utils.sheet_to_json<(string | number | null)[]>(sheet, { header: 1, defval: null })
  const head = rows.slice(0, 8).flat().map((c) => String(c ?? "")).join(" ")
  if (/СБЕРБАНК/i.test(head)) return "sber"
  if (/Банк ПСБ|ПРОМСВЯЗ/i.test(head)) return "psb"
  return "vtb"
}

export function parseStatement(format: BankFormat, workbook: XLSX.WorkBook): ParseResult {
  const transactions =
    format === "vtb" ? parseVtbStatement(workbook)
    : format === "psb" ? parsePsbStatement(workbook)
    : parseSberStatement(workbook)
  return { format, transactions }
}
```

    **КРИТИЧНО — порядок и enum:**
    1. `ALTER TYPE "ERP_SECTION" ADD VALUE IF NOT EXISTS 'CREDITS';`
       (Postgres требует commit между ADD VALUE и использованием — но т.к. CREDITS нигде в этой миграции в DEFAULT не используется, это безопасно одной транзакцией; в проде `prisma migrate deploy` каждый файл = отдельная транзакция.)

    2. **CREATE TABLE "Bank"**:
       ```sql
       CREATE TABLE "Bank" (
         "id" TEXT NOT NULL,
         "name" TEXT NOT NULL,
         "sortOrder" INTEGER NOT NULL DEFAULT 0,
         "createdAt" TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
         "updatedAt" TIMESTAMP(3) NOT NULL,
         CONSTRAINT "Bank_pkey" PRIMARY KEY ("id")
       );
       CREATE UNIQUE INDEX "Bank_name_key" ON "Bank"("name");
       ```

    3. **CREATE TABLE "Loan"**:
       - поля: id TEXT PK, contractNumber TEXT NOT NULL, companyId TEXT NOT NULL, bankId TEXT NOT NULL, amount DECIMAL(14,2) NOT NULL, annualRatePct DECIMAL(6,3) NOT NULL, termMonths INTEGER, issueDate TIMESTAMP(3), notes TEXT, deletedAt TIMESTAMP(3), createdAt TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP, updatedAt TIMESTAMP(3) NOT NULL
       - indexes: `Loan_deletedAt_idx`, `Loan_companyId_idx`, `Loan_bankId_idx`
       - FK: `Loan_companyId_fkey` → "Company"("id") ON DELETE RESTRICT ON UPDATE CASCADE; `Loan_bankId_fkey` → "Bank"("id") ON DELETE RESTRICT ON UPDATE CASCADE

    4. **CREATE TABLE "LoanPayment"**:
       - поля: id TEXT PK, loanId TEXT NOT NULL, date DATE NOT NULL, principal DECIMAL(14,2) NOT NULL DEFAULT 0, interest DECIMAL(14,2) NOT NULL DEFAULT 0, createdAt TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP
       - index: `CREATE INDEX "LoanPayment_loanId_date_idx" ON "LoanPayment"("loanId", "date");`
       - FK: `LoanPayment_loanId_fkey` → "Loan"("id") ON DELETE CASCADE ON UPDATE CASCADE

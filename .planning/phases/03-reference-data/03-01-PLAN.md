From prisma/schema.prisma — relevant models:
```prisma
model Brand {
  id         String     @id @default(cuid())
  name       String     @unique
  categories Category[]
  products   Product[]
}

model Category {
  id            String        @id @default(cuid())
  name          String
  brandId       String
  brand         Brand         @relation(fields: [brandId], references: [id], onDelete: Restrict)
  subcategories Subcategory[]
  products      Product[]
  @@unique([name, brandId])
  -- Prisma generates: where: { name_brandId: { name, brandId } }
}

model Subcategory {
  id         String    @id @default(cuid())
  name       String
  categoryId String
  category   Category  @relation(fields: [categoryId], references: [id], onDelete: Cascade)
  products   Product[]
  @@unique([name, categoryId])
  -- Prisma generates: where: { name_categoryId: { name, categoryId } }
}

model Marketplace {
  id       String               @id @default(cuid())
  name     String               @unique
  slug     String               @unique
  articles MarketplaceArticle[]
}
```

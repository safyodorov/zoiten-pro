# Plan 17-01 — Foundation

## Goal

Положить фундамент: схема БД, миграция с FK и indexes, расширение `lib/wb-api.ts` для парсинга `characteristics` + `sizes[].techSize`, интеграция в `/api/wb-sync` чтобы новые поля WbCard заполнялись на проде.

## Tasks

### T1: Prisma schema changes

**File:** `prisma/schema.prisma`

```prisma
// Расширить ProductDirection (одна новая строка)
model ProductDirection {
  ... existing ...
  hasSizes  Boolean  @default(false)
  ... existing ...
}

// Расширить WbCard (две новые строки)
model WbCard {
  ... existing ...
  characteristics Json?
  techSizes       String[] @default([])
  ... existing ...
}

// Новый enum
enum PropertyKind {
  STRING
  ENUM
  NUMBER
}

// Новая модель (после Subcategory)
model CategoryProperty {
  id         String                 @id @default(cuid())
  categoryId String
  category   Category               @relation(fields: [categoryId], references: [id], onDelete: Cascade)
  name       String
  kind       PropertyKind           @default(STRING)
  options    String[]               @default([])
  wbAttrName String?
  sortOrder  Int                    @default(0)
  values     ProductPropertyValue[]
  createdAt  DateTime               @default(now())
  updatedAt  DateTime               @updatedAt
  @@unique([categoryId, name])
}

// Новая модель (рядом)
model ProductPropertyValue {
  id         String           @id @default(cuid())
  productId  String
  product    Product          @relation(fields: [productId], references: [id], onDelete: Cascade)
  propertyId String
  property   CategoryProperty @relation(fields: [propertyId], references: [id], onDelete: Cascade)
  value      String
  createdAt  DateTime         @default(now())
  updatedAt  DateTime         @updatedAt
  @@unique([productId, propertyId])
  @@index([propertyId])
}

// Новая модель (рядом)
model ProductSize {
  id        String   @id @default(cuid())
  productId String
  product   Product  @relation(fields: [productId], references: [id], onDelete: Cascade)
  value     String
  sortOrder Int      @default(0)
  createdAt DateTime @default(now())
  @@unique([productId, value])
  @@index([productId])
}

// Расширить Product (back-relations)
model Product {
  ... existing ...
  propertyValues ProductPropertyValue[]
  sizes          ProductSize[]
  ... existing ...
}
```

### T2: Миграция SQL

**File:** `prisma/migrations/20260511_phase17_properties_sizes/migration.sql`

```sql
-- Phase 17: Свойства товаров + размерная сетка
-- D-01 свойства → к Категории, D-02 hasSizes → к Направлению, D-04 размер = отдельная сущность

-- 1) Расширение ProductDirection
ALTER TABLE "ProductDirection" ADD COLUMN "hasSizes" BOOLEAN NOT NULL DEFAULT false;
UPDATE "ProductDirection" SET "hasSizes" = true WHERE "name" = 'Одежда';

-- 2) Расширение WbCard
ALTER TABLE "WbCard" ADD COLUMN "characteristics" JSONB;
ALTER TABLE "WbCard" ADD COLUMN "techSizes" TEXT[] NOT NULL DEFAULT '{}';

-- 3) Enum PropertyKind
CREATE TYPE "PropertyKind" AS ENUM ('STRING', 'ENUM', 'NUMBER');

-- 4) CategoryProperty
CREATE TABLE "CategoryProperty" (
    "id" TEXT NOT NULL,
    "categoryId" TEXT NOT NULL,
    "name" TEXT NOT NULL,
    "kind" "PropertyKind" NOT NULL DEFAULT 'STRING',
    "options" TEXT[] NOT NULL DEFAULT '{}',
    "wbAttrName" TEXT,
    "sortOrder" INTEGER NOT NULL DEFAULT 0,
    "createdAt" TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "updatedAt" TIMESTAMP(3) NOT NULL,
    CONSTRAINT "CategoryProperty_pkey" PRIMARY KEY ("id")
);
CREATE UNIQUE INDEX "CategoryProperty_categoryId_name_key" ON "CategoryProperty"("categoryId", "name");
ALTER TABLE "CategoryProperty" ADD CONSTRAINT "CategoryProperty_categoryId_fkey"
    FOREIGN KEY ("categoryId") REFERENCES "Category"("id") ON DELETE CASCADE ON UPDATE CASCADE;

-- 5) ProductPropertyValue
CREATE TABLE "ProductPropertyValue" (
    "id" TEXT NOT NULL,
    "productId" TEXT NOT NULL,
    "propertyId" TEXT NOT NULL,
    "value" TEXT NOT NULL,
    "createdAt" TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "updatedAt" TIMESTAMP(3) NOT NULL,
    CONSTRAINT "ProductPropertyValue_pkey" PRIMARY KEY ("id")
);
CREATE UNIQUE INDEX "ProductPropertyValue_productId_propertyId_key" ON "ProductPropertyValue"("productId", "propertyId");
CREATE INDEX "ProductPropertyValue_propertyId_idx" ON "ProductPropertyValue"("propertyId");
ALTER TABLE "ProductPropertyValue" ADD CONSTRAINT "ProductPropertyValue_productId_fkey"
    FOREIGN KEY ("productId") REFERENCES "Product"("id") ON DELETE CASCADE ON UPDATE CASCADE;
ALTER TABLE "ProductPropertyValue" ADD CONSTRAINT "ProductPropertyValue_propertyId_fkey"
    FOREIGN KEY ("propertyId") REFERENCES "CategoryProperty"("id") ON DELETE CASCADE ON UPDATE CASCADE;

-- 6) ProductSize
CREATE TABLE "ProductSize" (
    "id" TEXT NOT NULL,
    "productId" TEXT NOT NULL,
    "value" TEXT NOT NULL,
    "sortOrder" INTEGER NOT NULL DEFAULT 0,
    "createdAt" TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT "ProductSize_pkey" PRIMARY KEY ("id")
);
CREATE UNIQUE INDEX "ProductSize_productId_value_key" ON "ProductSize"("productId", "value");
CREATE INDEX "ProductSize_productId_idx" ON "ProductSize"("productId");
ALTER TABLE "ProductSize" ADD CONSTRAINT "ProductSize_productId_fkey"
    FOREIGN KEY ("productId") REFERENCES "Product"("id") ON DELETE CASCADE ON UPDATE CASCADE;
```

### T3: WB API parser extension

**File:** `lib/wb-api.ts` (точное место — функция парсинга `/content/v2/get/cards/list` response)

Найти где сейчас собирается WbCard payload (поля `name`, `brand`, `photos`, `weightKg`...). Добавить:

```typescript
// Из card.characteristics — массив { id, name, value }
// value может быть string или string[]
characteristics: card.characteristics ?? null,
// Из card.sizes — массив { techSize, wbSize, ... }
techSizes: (card.sizes ?? [])
  .map((s: any) => String(s.techSize ?? '').trim())
  .filter(Boolean),
```

**Защита:** если `card.characteristics` отсутствует → null, не пустой массив, чтобы можно было различить «парсинг не делался» и «свойств нет».

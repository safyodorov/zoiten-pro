    ```typescript
    // Phase 19: Pure helpers для chart timeseries по WbAdvertStatDaily.
    // Параллельно с lib/wb-orders-chart.ts (которая для /cards/wb).
    // 28 точек, MSK даты, zero-fill, ДРР с zero-guard.

    import { getLast28DaysMsk } from "@/lib/wb-orders-chart"

    export interface AdvertDayPoint {
      date: string      // YYYY-MM-DD
      spent: number     // ₽
      orders: number    // шт
      drr: number | null // %
    }

    export interface AdvertRawPoint {
      date: string
      sum: number       // ₽ потрачено
      orders: number
      sumPrice: number  // ₽ оборот
    }

    /** Раскладывает raw rows в 28 точек, zero-fill пропусков. */
    export function fillAdvertTimeSeries(
      raw: AdvertRawPoint[],
      now: Date = new Date(),
    ): AdvertDayPoint[] {
      const dates = getLast28DaysMsk(now)
      const byDate = new Map<string, { sum: number; orders: number; sumPrice: number }>()
      for (const r of raw) {
        const acc = byDate.get(r.date) ?? { sum: 0, orders: 0, sumPrice: 0 }
        acc.sum += r.sum
        acc.orders += r.orders
        acc.sumPrice += r.sumPrice
        byDate.set(r.date, acc)
      }
      return dates.map(d => {
        const v = byDate.get(d)
        return {
          date: d,
          spent: v?.sum ?? 0,
          orders: v?.orders ?? 0,
          drr: v && v.sumPrice > 0 ? (v.sum / v.sumPrice) * 100 : null,
        }
      })
    }
    ```

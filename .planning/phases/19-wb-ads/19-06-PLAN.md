    ```typescript
    // Phase 19: Pure helpers для chart timeseries по WbAdvertStatDaily.
    // Параллельно с lib/wb-orders-chart.ts (которая для /cards/wb).
    // 28 точек, MSK даты, zero-fill, ДРР с zero-guard.

    import { getLast28DaysMsk } from "@/lib/wb-orders-chart"

    export interface AdvertDayPoint {
      date: string      // YYYY-MM-DD
      spent: number     // ₽
      orders: number    // шт
      drr: number | null // %
    }

    export interface AdvertRawPoint {
      date: string
      sum: number       // ₽ потрачено
      orders: number
      sumPrice: number  // ₽ оборот
    }

    /** Раскладывает raw rows в 28 точек, zero-fill пропусков. */
    export function fillAdvertTimeSeries(
      raw: AdvertRawPoint[],
      now: Date = new Date(),
    ): AdvertDayPoint[] {
      const dates = getLast28DaysMsk(now)
      const byDate = new Map<string, { sum: number; orders: number; sumPrice: number }>()
      for (const r of raw) {
        const acc = byDate.get(r.date) ?? { sum: 0, orders: 0, sumPrice: 0 }
        acc.sum += r.sum
        acc.orders += r.orders
        acc.sumPrice += r.sumPrice
        byDate.set(r.date, acc)
      }
      return dates.map(d => {
        const v = byDate.get(d)
        return {
          date: d,
          spent: v?.sum ?? 0,
          orders: v?.orders ?? 0,
          drr: v && v.sumPrice > 0 ? (v.sum / v.sumPrice) * 100 : null,
        }
      })
    }
    ```

    ```typescript
    export function WbAdvertExpandPanel({ group }: Props) {
      const { nmIdAgg, imtIdAgg, nmIdToImtId, ordersCharts, campaigns } = group
      if (!nmIdAgg || !nmIdToImtId) return null  // должно быть из Plan 19-05

      // Группировка nmIds по imtId (null → "no-imt" bucket)
      const NO_IMT = "no-imt" as const
      const nmsByImt = new Map<number | typeof NO_IMT, number[]>()
      for (const [nm, imt] of nmIdToImtId) {
        const key = imt ?? NO_IMT
        const list = nmsByImt.get(key) ?? []
        list.push(nm); nmsByImt.set(key, list)
      }

      // Подсчёт кампаний per imtId (campaigns, у которых хотя бы 1 target в этой imt-склейке)
      const campaignsByImt = new Map<number | typeof NO_IMT, number>()
      for (const c of campaigns) {
        const seenImts = new Set<number | typeof NO_IMT>()
        for (const [nm, imt] of nmIdToImtId) {
          // grubo: для целей UI считаем что campaign принадлежит imt если хотя бы 1 nmId совпадает
          if (group.product && nm) {
            const key = imt ?? NO_IMT
            seenImts.add(key)
          }
        }
        for (const k of seenImts) campaignsByImt.set(k, (campaignsByImt.get(k) ?? 0) + 1)
      }
      // Более простой подсчёт — campaignCount = campaigns.length для всех imt (для каждой imt продукта те же кампании в большинстве случаев). Уточнить при имплементации.

      const chartByNmId = new Map<number, AdvertDayPoint[]>()
      for (const oc of ordersCharts ?? []) chartByNmId.set(oc.nmId, oc.data)

      const imtKeys = Array.from(nmsByImt.keys())
      // Сортировка: numeric imt по убыванию totalSpent, NO_IMT в конец
      imtKeys.sort((a, b) => {
        if (a === NO_IMT) return 1
        if (b === NO_IMT) return -1
        const aAgg = imtIdAgg?.get(a as number)?.totalSpent ?? 0
        const bAgg = imtIdAgg?.get(b as number)?.totalSpent ?? 0
        return bAgg - aAgg
      })

      return (
        <div className="flex flex-col gap-3 p-4 bg-muted/20">
          {imtKeys.map(imtKey => {
            const nms = nmsByImt.get(imtKey)!
            const isImt = imtKey !== NO_IMT
            const imtAgg = isImt ? imtIdAgg?.get(imtKey as number) : null
            return (
              <div key={String(imtKey)} className="flex flex-col gap-2">
                {isImt && imtAgg && (
                  <PerImtBlock
                    imtId={imtKey as number}
                    productCount={nms.length}
                    campaignCount={campaigns.length}
                    agg={imtAgg}
                  />
                )}
                {!isImt && (
                  <div className="px-3 py-1.5 text-xs text-muted-foreground italic">

                  </div>
                )}

                      nmId={nm}
                      chart={chartByNmId.get(nm) ?? null}

                        drr: null, cpc: null, ctr: null, cr: null,
                      }}
                    />
                  ))}
                </div>
              </div>
            )
          })}
        </div>
      )
    }
    ```

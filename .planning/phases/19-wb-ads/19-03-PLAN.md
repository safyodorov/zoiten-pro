    ```typescript
    // Phase 19: WB Advert API client.
    // Endpoints: advert-api.wildberries.ru/adv/{v1,v2}/...
    // Rate limits: /promotion/* 5 req/sec, /fullstats 1 req/sec batch ≤100.
    // Cooldown bus: bucket 'advert' (изолирован от WB_API_TOKEN scope).
    // Token: WB_ADS_TOKEN (scope bit подтверждён в 19-W0-NOTES.md).

    import { getWbToken } from "@/lib/wb-token"
    import {
      getWbCooldownSecondsRemaining,
      setWbCooldownUntil,
    } from "@/lib/wb-cooldown"
    import { WbRateLimitError } from "@/lib/wb-api"

    const BASE_URL = "https://advert-api.wildberries.ru"
    const FULLSTATS_BATCH_SIZE = 100
    const PROMOTION_RATE_SLEEP_MS = 200      // 5 req/sec
    const FULLSTATS_RATE_SLEEP_MS = 1000     // 1 req/sec

    // ── Types ─────────────────────────────────────────────────────────

    export interface WbAdvertCount {
      advertId: number
      type: number
      status: number
      changeTime: string             // ISO
    }

    export interface WbAdvertInfo {
      advertId: number
      name: string | null
      type: number
      status: number
      dailyBudget: number | null
      startTime: string | null       // ISO
      endTime: string | null
      changeTime: string
      nmIds: number[]                // развёрнутые targets из params[].nms[]
      raw: unknown                   // полный ответ для аудита
    }

    export interface WbAdvertStat {
      advertId: number
      date: string                   // YYYY-MM-DD
      nmId: number
      appType: number
      views: number
      clicks: number
      ctr: number | null
      cpc: number | null
      sum: number
      atbs: number
      orders: number
      cr: number | null
      shks: number
      sumPrice: number
    }

    export interface WbAdvertBalance {
      balance: number
      bonus: number
      net: number | null
    }

    // ── Helpers ───────────────────────────────────────────────────────

    function sleep(ms: number): Promise<void> {
      return new Promise(r => setTimeout(r, ms))
    }

    async function callAdvert(
      path: string,
      init: RequestInit = {},
    ): Promise<Response> {
      const cooldown = await getWbCooldownSecondsRemaining("advert")
      if (cooldown > 0) {
        throw new WbRateLimitError(`Advert API (path=${path})`, cooldown)
      }
      const token = await getWbToken("WB_ADS_TOKEN")
      const res = await fetch(`${BASE_URL}${path}`, {
        ...init,
        headers: {
          Authorization: token,
          ...(init.body && !(init.body instanceof FormData)
            ? { "Content-Type": "application/json" }
            : {}),
          ...init.headers,
        },
      })
      if (res.status === 429) {
        const retryAfter = parseInt(res.headers.get("Retry-After") ?? "60", 10)
        await setWbCooldownUntil("advert", retryAfter)
        throw new WbRateLimitError(`Advert API ${path}`, retryAfter)
      }
      if (!res.ok) {
        const text = await res.text().catch(() => "")
        throw new Error(`Advert API ${path} → HTTP ${res.status}: ${text}`)
      }
      return res
    }

    // ── Public API ────────────────────────────────────────────────────

    /** GET /adv/v1/promotion/count → плоский список (advertId, type, status, changeTime) */
    export async function fetchPromotionCount(): Promise<WbAdvertCount[]> {
      await sleep(PROMOTION_RATE_SLEEP_MS)
      const res = await callAdvert("/adv/v1/promotion/count")
      const data = await res.json() as {
        adverts: Array<{
          type: number
          status: number
          count: number
          advert_list: Array<{ advertId: number; changeTime: string }>
        }>
      }
      const out: WbAdvertCount[] = []
      for (const group of data.adverts ?? []) {
        for (const item of group.advert_list ?? []) {
          out.push({
            advertId: item.advertId,
            type: group.type,
            status: group.status,
            changeTime: item.changeTime,
          })
        }
      }
      return out
    }

    /** POST /adv/v1/promotion/adverts с body [advertId, ...] → детали кампаний (name, daily budget, nmIds) */
    export async function fetchPromotionAdverts(
      advertIds: number[],
    ): Promise<WbAdvertInfo[]> {
      if (advertIds.length === 0) return []
      await sleep(PROMOTION_RATE_SLEEP_MS)
      const res = await callAdvert("/adv/v1/promotion/adverts", {
        method: "POST",
        body: JSON.stringify(advertIds),
      })
      const data = await res.json() as Array<{
        advertId: number
        name?: string
        type: number
        status: number
        dailyBudget?: number
        startTime?: string
        endTime?: string
        changeTime: string
        params?: Array<{ nms?: Array<{ nm: number }>; subject?: unknown }>
        autoParams?: { nms?: number[] }
      }>
      return data.map(r => ({
        advertId: r.advertId,
        name: r.name ?? null,
        type: r.type,
        status: r.status,
        dailyBudget: r.dailyBudget ?? null,
        startTime: r.startTime ?? null,
        endTime: r.endTime ?? null,
        changeTime: r.changeTime,
        nmIds: extractNmIds(r),
        raw: r,
      }))
    }

    function extractNmIds(r: {
      params?: Array<{ nms?: Array<{ nm: number }> }>
      autoParams?: { nms?: number[] }
    }): number[] {
      const ids = new Set<number>()
      for (const p of r.params ?? []) {
        for (const nm of p.nms ?? []) {
          if (nm.nm) ids.add(nm.nm)
        }
      }
      for (const id of r.autoParams?.nms ?? []) ids.add(id)
      return Array.from(ids)
    }

    /** POST /adv/v2/fullstats — батчами по 100 advertId, sleep 1 сек между. */
    export async function fetchFullStats(
      advertIds: number[],
      interval: { begin: string; end: string },  // 'YYYY-MM-DD'
    ): Promise<WbAdvertStat[]> {
      const all: WbAdvertStat[] = []
      for (let i = 0; i < advertIds.length; i += FULLSTATS_BATCH_SIZE) {
        if (i > 0) await sleep(FULLSTATS_RATE_SLEEP_MS)
        const batch = advertIds.slice(i, i + FULLSTATS_BATCH_SIZE)
        const body = batch.map(id => ({ id, interval }))
        const res = await callAdvert("/adv/v2/fullstats", {
          method: "POST",
          body: JSON.stringify(body),
        })
        const data = await res.json() as Array<{
          advertId: number
          days: Array<{
            date: string
            apps: Array<{
              appType: number
              views: number
              clicks: number
              ctr?: number
              cpc?: number
              sum: number
              atbs: number
              orders: number
              cr?: number
              shks: number
              sum_price: number
              nm: Array<{
                nmId: number
                views: number
                clicks: number
                ctr?: number
                cpc?: number
                sum: number
                atbs: number
                orders: number
                cr?: number
                shks: number
                sum_price: number
              }>
            }>
          }>
        }>
        for (const camp of data ?? []) {
          for (const day of camp.days ?? []) {
            // date в виде 'YYYY-MM-DD HH:MM:SS' — обрезаем
            const dateStr = day.date.slice(0, 10)
            for (const app of day.apps ?? []) {
              for (const nm of app.nm ?? []) {
                all.push({
                  advertId: camp.advertId,
                  date: dateStr,
                  nmId: nm.nmId,
                  appType: app.appType,
                  views: nm.views,
                  clicks: nm.clicks,
                  ctr: nm.ctr ?? null,
                  cpc: nm.cpc ?? null,
                  sum: nm.sum,
                  atbs: nm.atbs,
                  orders: nm.orders,
                  cr: nm.cr ?? null,
                  shks: nm.shks,
                  sumPrice: nm.sum_price,
                })
              }
            }
          }
        }
      }
      return all
    }

    /** GET /adv/v1/balance → текущий баланс рекламного счёта */
    export async function fetchBalance(): Promise<WbAdvertBalance> {
      await sleep(PROMOTION_RATE_SLEEP_MS)
      const res = await callAdvert("/adv/v1/balance")
      const data = await res.json() as {
        balance: number
        bonus: number
        net?: number
      }
      return {
        balance: data.balance,
        bonus: data.bonus,
        net: data.net ?? null,
      }
    }
    ```

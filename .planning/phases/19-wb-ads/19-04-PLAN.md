Реализовать daily cron + manual backfill для синхронизации рекламных данных WB. Cron-endpoint оркестрирует все 4 функции из `lib/wb-adv-api.ts` (Plan 19-03), upsert'ит в 4 таблицы (Plan 19-01), пишет lastRun marker, обрабатывает 429 через WbRateLimitError. Purpose: Без cron'a у нас нет автоматической синхронизации — UI (Plan 19-05) будет показывать пустые данные. Backfill нужен для первого запуска и для catch-up если что-то отвалилось.

    ```typescript
    // Phase 19: Daily cron оркестрации WB Advert sync.
    // Источник истины — WB Advert API. Защищён x-cron-secret.
    // Идемпотентный — повторный run перезаписывает upsert'ом, не создаёт дубликаты.

    import { NextRequest, NextResponse } from "next/server"
    import { prisma } from "@/lib/prisma"
    import {
      fetchPromotionCount,
      fetchPromotionAdverts,
      fetchFullStats,
      fetchBalance,
    } from "@/lib/wb-adv-api"
    import { WbRateLimitError } from "@/lib/wb-api"
    import { getMskTodayDate } from "@/lib/wb-orders-chart"
    import { getMskTodayString } from "@/lib/wb-cron-schedule"

    export const runtime = "nodejs"
    export const maxDuration = 600

    const DAILY_DELTA_DAYS = 7

    function formatDate(d: Date): string {
      return d.toISOString().slice(0, 10)  // YYYY-MM-DD
    }

    /** Выполняет полную orchestration. Возвращает счётчики. */
    export async function runAdvSync(daysWindow: number = DAILY_DELTA_DAYS) {
      // 1. Список кампаний
      const campaigns = await fetchPromotionCount()
      console.log(`[wb-adv-sync] fetched ${campaigns.length} campaigns`)

      // 2. Upsert WbAdvertCampaign (минимальные поля)
      for (const c of campaigns) {
        await prisma.wbAdvertCampaign.upsert({
          where: { advertId: c.advertId },
          create: {
            advertId: c.advertId,
            type: c.type,
            status: c.status,
            changeTime: new Date(c.changeTime),
          },
          update: {
            type: c.type,
            status: c.status,
            changeTime: new Date(c.changeTime),
          },
        })
      }

      // 3. Детали для каждой advertId (батчами по 50)
      const advertIds = campaigns.map(c => c.advertId)
      const detailBatchSize = 50
      let detailsUpdated = 0
      for (let i = 0; i < advertIds.length; i += detailBatchSize) {
        const batch = advertIds.slice(i, i + detailBatchSize)
        const details = await fetchPromotionAdverts(batch)
        for (const d of details) {
          await prisma.wbAdvertCampaign.update({
            where: { advertId: d.advertId },
            data: {
              name: d.name,
              dailyBudget: d.dailyBudget,
              startDate: d.startTime ? new Date(d.startTime) : null,
              endDate: d.endTime ? new Date(d.endTime) : null,
              raw: d.raw as never,
            },
          })
          // Upsert targets — текущие nmIds. Прежние неактивные не удаляем
          // (active flag используется только для UI фильтра — текущая логика: всё что приходит = active).
          for (const nmId of d.nmIds) {
            await prisma.wbAdvertTarget.upsert({
              where: { advertId_nmId: { advertId: d.advertId, nmId } },
              create: { advertId: d.advertId, nmId, active: true },
              update: { active: true },
            })
          }
          detailsUpdated++
        }
      }
      console.log(`[wb-adv-sync] updated ${detailsUpdated} campaign details + targets`)

      // 4. Статистика rolling N дней
      const today = getMskTodayDate()
      const begin = new Date(today.getTime() - daysWindow * 24 * 3600_000)
      const end = new Date(today.getTime() - 24 * 3600_000) // вчера
      const activeAdvertIds = campaigns
        .filter(c => c.status === 9 || c.status === 11)  // active / paused
        .map(c => c.advertId)
      const stats = activeAdvertIds.length > 0
        ? await fetchFullStats(activeAdvertIds, {
            begin: formatDate(begin),
            end: formatDate(end),
          })
        : []
      console.log(`[wb-adv-sync] fetched ${stats.length} stat rows`)

      // 5. Upsert WbAdvertStatDaily (compound unique)
      let statsUpserted = 0
      for (const s of stats) {
        await prisma.wbAdvertStatDaily.upsert({
          where: {
            advertId_date_nmId_appType: {
              advertId: s.advertId,
              date: new Date(s.date),
              nmId: s.nmId,
              appType: s.appType,
            },
          },
          create: {
            advertId: s.advertId,
            date: new Date(s.date),
            nmId: s.nmId,
            appType: s.appType,
            views: s.views,
            clicks: s.clicks,
            ctr: s.ctr,
            cpc: s.cpc,
            sum: s.sum,
            atbs: s.atbs,
            orders: s.orders,
            cr: s.cr,
            shks: s.shks,
            sumPrice: s.sumPrice,
          },
          update: {
            views: s.views,
            clicks: s.clicks,
            ctr: s.ctr,
            cpc: s.cpc,
            sum: s.sum,
            atbs: s.atbs,
            orders: s.orders,
            cr: s.cr,
            shks: s.shks,
            sumPrice: s.sumPrice,
          },
        })
        statsUpserted++
      }

      // 6. Balance snapshot — всегда create (история)
      const balance = await fetchBalance()
      await prisma.wbAdvertBalanceSnapshot.create({
        data: {
          balance: balance.balance,
          bonus: balance.bonus,
          net: balance.net,
        },
      })

      return {
        campaigns: campaigns.length,
        detailsUpdated,
        statsUpserted,
        balance: balance.balance,
      }
    }

    export async function GET(req: NextRequest): Promise<NextResponse> {
      const secret = req.headers.get("x-cron-secret")
      if (!secret || secret !== process.env.CRON_SECRET) {
        return NextResponse.json({ error: "Не авторизован" }, { status: 401 })
      }
      try {
        console.log(`[wb-adv-sync cron] start`)
        const result = await runAdvSync(DAILY_DELTA_DAYS)
        console.log(`[wb-adv-sync cron] done`, result)
        const todayStr = getMskTodayString()
        await prisma.appSetting.upsert({
          where: { key: "wbAdvSyncLastRun" },
          create: { key: "wbAdvSyncLastRun", value: todayStr },
          update: { value: todayStr },
        })
        return NextResponse.json({ ok: true, ...result })
      } catch (err) {
        if (err instanceof WbRateLimitError) {
          return NextResponse.json(
            { ok: false, error: "rate-limit", retryAfterSec: err.retryAfterSec },
            { status: 429 },
          )
        }
        console.error("[wb-adv-sync cron] error:", err)
        return NextResponse.json(
          { ok: false, error: (err as Error).message },
          { status: 500 },
        )
      }
    }
    ```

    2. Создать `app/api/wb-adv-backfill/route.ts` (POST ONLY — без GET alias; backfill вызывается только через `curl -X POST` с VPS):

    ```typescript
    // Phase 19: Manual backfill WB Advert за окно N дней (1..30).
    // x-cron-secret guard (для consistency с /api/wb-orders-backfill — оба используются
    // и из UI кнопки через server action, и через curl на VPS).
    // POST-only — никакого GET alias, чтобы не ломать Next.js handler conventions
    // и не разрешать accidental triggering через browser.

    import { NextRequest, NextResponse } from "next/server"
    import { runAdvSync } from "@/app/api/cron/wb-adv-sync/route"
    import { WbRateLimitError } from "@/lib/wb-api"

    export const runtime = "nodejs"
    export const maxDuration = 600

    export async function POST(req: NextRequest): Promise<NextResponse> {
      const secret = req.headers.get("x-cron-secret")
      if (!secret || secret !== process.env.CRON_SECRET) {
        return NextResponse.json({ error: "Не авторизован" }, { status: 401 })
      }
      const { searchParams } = new URL(req.url)
      const daysParam = searchParams.get("days")
      const days = daysParam ? parseInt(daysParam, 10) : 7
      if (!Number.isFinite(days) || days < 1 || days > 30) {
        return NextResponse.json(
          { ok: false, error: "Параметр days должен быть целым числом 1..30" },
          { status: 400 },
        )
      }
      try {
        console.log(`[wb-adv-backfill] start days=${days}`)
        const result = await runAdvSync(days)
        console.log(`[wb-adv-backfill] done`, result)
        return NextResponse.json({ ok: true, days, ...result })
      } catch (err) {
        if (err instanceof WbRateLimitError) {
          return NextResponse.json(
            { ok: false, error: "rate-limit", retryAfterSec: err.retryAfterSec },
            { status: 429 },
          )
        }
        console.error("[wb-adv-backfill] error:", err)
        return NextResponse.json(
          { ok: false, error: (err as Error).message },
          { status: 500 },
        )
      }
    }

    // NB: НЕТ `export const GET = POST` — backfill дёргается только через POST с VPS curl.
    // Если в будущем понадобится GET-вариант, нужно создать отдельный handler с своими
    // валидациями (GET не должен изменять состояние, а backfill пишет в БД).
    ```

    Замечание: `runAdvSync` экспортируется из route.ts чтобы backfill мог его переиспользовать. Это валидный паттерн в Next.js App Router — route.ts может экспортировать helpers наряду с GET/POST.

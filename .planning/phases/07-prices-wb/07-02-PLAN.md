/** Рассчитать юнит-экономику одной ценовой строки WbCard.
 *  Pure function — детерминированная, без side effects.
 *
 *  Формулы выведены из canonical Excel "Форма управления ценами.xlsx".
 *  Golden test (nmId 800750522): profit ≈ 567.68 ₽, returnOnSales ≈ 7%, ROI ≈ 26%.
 */
export function calculatePricing(inputs: PricingInputs): PricingOutputs {
  // Guard against negative / zero edge cases
  const priceBeforeDiscount = Math.max(0, inputs.priceBeforeDiscount)
  const costPrice = Math.max(0, inputs.costPrice)

  // ── 1. Цена продавца (после скидки продавца) ───────────────
  const sellerPrice =
    priceBeforeDiscount * (1 - inputs.sellerDiscountPct / 100)

  // ── 2. Цена со скидкой WB (СПП) ─────────────────────────────
  const priceAfterWbDiscount =
    sellerPrice * (1 - inputs.wbDiscountPct / 100)

  // ── 3. Цена со скидкой клуба WB ─────────────────────────────
  const priceAfterClubDiscount =
    priceAfterWbDiscount * (1 - inputs.clubDiscountPct / 100)

  // ── 4. Кошелёк (покупательская программа) ───────────────────
  // EXECUTOR: уточнить по Excel — кошелёк считается от priceAfterWbDiscount или priceAfterClubDiscount
  const walletAmount = priceAfterClubDiscount * (inputs.walletPct / 100)
  const priceAfterWallet = priceAfterClubDiscount - walletAmount

  // ── 5. Эквайринг (% от priceAfterWallet) ────────────────────
  const acquiringAmount = priceAfterWallet * (inputs.acquiringPct / 100)

  // ── 6. Комиссия ИУ FBW (% от priceAfterWbDiscount) ──────────
  // EXECUTOR: уточнить точную базу комиссии из Excel. Если profit != 567.68 — попробовать другую базу.
  const commissionAmount = priceAfterWbDiscount * (inputs.commFbwPct / 100)

  // ── 7. ДРР (% от sellerPrice) ───────────────────────────────
  const drrAmount = sellerPrice * (inputs.drrPct / 100)

  // ── 8. Джем (% от sellerPrice — из Excel ~77.5 при базе 7749.9) ──
  const jemAmount = sellerPrice * (inputs.jemPct / 100)

  // ── 9. К перечислению (доход продавца до расходов) ──────────
  // К перечислению = priceAfterWallet - эквайринг - комиссия - ДРР - джем
  const transferAmount =
    priceAfterWallet - acquiringAmount - commissionAmount - drrAmount - jemAmount

  // ── 10. Брак (% от себестоимости) ───────────────────────────
  const defectAmount = costPrice * (inputs.defectRatePct / 100)

  // ── 11. Доставка (фиксированная сумма из Product.deliveryCostRub) ──
  const deliveryAmount = inputs.deliveryCostRub

  // ── 12. Кредит (% от sellerPrice или priceBeforeDiscount? — из Excel 232.5) ──

  // ── 14. Налог (% от sellerPrice, проверено: 8% × 7749.9 = 619.992 ≈ 620) ──
  const taxAmount = sellerPrice * (inputs.taxPct / 100)

  // ── 15. Прибыль ─────────────────────────────────────────────
  const profit =
    transferAmount - costPrice - defectAmount - deliveryAmount - creditAmount - overheadAmount - taxAmount

  // ── 16. Re продаж (%) — прибыль относительно цены продавца ──
  const returnOnSalesPct =
    sellerPrice > 0 ? (profit / sellerPrice) * 100 : 0

  // ── 17. ROI (%) — прибыль относительно вложений (costPrice) ──
  const roiPct =
    costPrice > 0 ? (profit / costPrice) * 100 : 0

  return {
    sellerPrice,
    priceAfterWbDiscount,
    priceAfterClubDiscount,
    priceAfterWallet,
    walletAmount,
    acquiringAmount,
    commissionAmount,
    drrAmount,
    jemAmount,

  }
}

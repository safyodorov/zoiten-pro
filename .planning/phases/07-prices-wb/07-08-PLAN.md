<!-- Критические правила сборки данных: -->

1. **linkedNmIds расчёт:**
   - `MarketplaceArticle` с `slug="wb"` и `product.deletedAt: null`
   - `nmIds = articles.map(a => parseInt(a.article, 10)).filter(n => !isNaN(n))`

2. **WbCards load:**
   - `prisma.wbCard.findMany({ where: { nmId: { in: linkedNmIds } } })`

3. **Promotions load (для активных акций):**
   - `prisma.wbPromotion.findMany({ where: { endDateTime: { gte: new Date() } }, include: { nomenclatures: true } })`
   - Активные = те которые ещё не истекли (endDateTime >= now)

4. **CalculatedPrice load:**
   - `prisma.calculatedPrice.findMany({ where: { wbCardId: { in: wbCardIds } } })`

5. **Group by Product:**
   - Для каждого Product строим ProductGroup
   - `product.id` → набор связанных WbCard через `MarketplaceArticle.article === String(card.nmId)`
   - Игнорировать `product.deletedAt != null` (soft-deleted)

6. **Строим priceRows для каждой WbCard:**

   a. **Текущая строка (current):**
   ```typescript
   const currentRow: PriceRow = {
     id: `${card.id}-current`,
     type: "current",
     sellerPriceBeforeDiscount: card.priceBeforeDiscount ?? 0,
     label: "Текущая",
     computed: calculatePricing({
       priceBeforeDiscount: card.priceBeforeDiscount ?? 0,
       sellerDiscountPct: card.sellerDiscount ?? 0,
       wbDiscountPct: card.discountWb ?? 0,
       clubDiscountPct: card.clubDiscount ?? 0,
       commFbwPct: card.commFbwIu ?? card.commFbwStd ?? 0,
       costPrice: product.cost?.costPrice ?? 0,
       buyoutPct: card.buyoutPercent ?? 100,
       drrPct: resolveDrrPct({
         productOverride: product.drrOverridePct,
         subcategoryDefault: product.subcategory?.defaultDrrPct ?? null,
       }),
       defectRatePct: resolveDefectRatePct({
         productOverride: product.defectRateOverridePct,
         categoryDefault: product.category?.defaultDefectRatePct ?? null,
       }),
       deliveryCostRub: resolveDeliveryCostRub(product.deliveryCostRub),
       walletPct: rates.wbWalletPct,
       acquiringPct: rates.wbAcquiringPct,
       jemPct: rates.wbJemPct,
       creditPct: rates.wbCreditPct,
       overheadPct: rates.wbOverheadPct,
       taxPct: rates.wbTaxPct,
     }),
   }
   ```

   b. **Regular promotions:** для каждой `WbPromotionNomenclature` с `promotionId` → `type="regular"` и `nmId === card.nmId`:
   - `sellerPriceBeforeDiscount: nom.planPrice ?? 0` (плановая цена акции)
   - `label: promotion.name`
   - `promotionDescription: promotion.description`
   - `promotionAdvantages: promotion.advantages`
   - Сортировка: DESC by `planPrice`

   c. **Auto promotions:** аналогично regular, но `type === "auto"` и фильтр — только те, у которых `planPrice !== null` (Excel данные загружены, D-10).
   - Сортировка: DESC by `planPrice`

   d. **Calculated prices:** для каждого `CalculatedPrice` с `wbCardId === card.id` отсортированных по `slot`:
   - `type: "calculated"`
   - `sellerPriceBeforeDiscount: cp.sellerPrice` (это уже цена продавца до скидки)
   - `label: cp.name`
   - `calculatedSlot: cp.slot as 1 | 2 | 3`
   - При расчёте используют resolved overrides с учётом `cp.drrPct`, `cp.defectRatePct`, `cp.deliveryCostRub` (если заданы — они override fallback chain для ЭТОЙ расчётки):
     - `drrPct: cp.drrPct ?? resolveDrrPct({...})`
     - `defectRatePct: cp.defectRatePct ?? resolveDefectRatePct({...})`
     - `deliveryCostRub: cp.deliveryCostRub ?? resolveDeliveryCostRub(product.deliveryCostRub)`

7. **ProductGroup построение:**

   ```typescript
   const group: ProductGroup = {

   ```

```tsx
// app/(dashboard)/prices/wb/page.tsx
// Phase 7: RSC страница раздела «Управление ценами → WB».
//
// Загружает все данные параллельно, группирует по Product,
// строит priceRows с серверным расчётом через calculatePricing,
// рендерит GlobalRatesBar + кнопки шапки + PriceCalculatorTable.

import { prisma } from "@/lib/prisma"
import { requireSection } from "@/lib/rbac"
import {
  calculatePricing,
  resolveDrrPct,
  resolveDefectRatePct,
  resolveDeliveryCostRub,
  type PricingInputs,
} from "@/lib/pricing-math"
import { GlobalRatesBar } from "@/components/prices/GlobalRatesBar"
import {
  PriceCalculatorTable,
  type ProductGroup,
  type PriceRow,
  type WbCardRowGroup,
} from "@/components/prices/PriceCalculatorTable"
import { WbSyncButton } from "@/components/cards/WbSyncButton"
import { WbSyncSppButton } from "@/components/cards/WbSyncSppButton"

// TODO (план 07-10): добавить импорты WbPromotionsSyncButton, WbAutoPromoUploadButton + Alert empty state

export const dynamic = "force-dynamic"

export default async function PricesWbPage() {
  // RBAC: view достаточно для чтения таблицы
  await requireSection("PRICES")

  // ── 1. Найти WB marketplace ─────────────────────────────────
  const wbMarketplace = await prisma.marketplace.findFirst({
    where: { slug: "wb" },
  })

  if (!wbMarketplace) {
    return (
      <div className="text-center py-16 text-muted-foreground">
        <p className="text-sm">
          WB marketplace не найден в справочнике. Проверьте настройки администратора.
        </p>
      </div>
    )
  }

  // ── 2. Параллельная загрузка данных ──────────────────────────
  const [appSettings, promotions, linkedArticles] = await Promise.all([
    // 6 глобальных ставок
    prisma.appSetting.findMany({
      where: {
        key: {
          in: [
            "wbWalletPct",
            "wbAcquiringPct",
            "wbJemPct",
            "wbCreditPct",
            "wbOverheadPct",
            "wbTaxPct",
          ],
        },
      },
    }),
    // Активные акции с номенклатурами
    prisma.wbPromotion.findMany({
      where: { endDateTime: { gte: new Date() } },
      include: { nomenclatures: true },
    }),
    // Привязанные к товарам статьи WB
    prisma.marketplaceArticle.findMany({
      where: {
        marketplaceId: wbMarketplace.id,
        product: { deletedAt: null },
      },
      include: {
        product: {
          include: {
            cost: true,
            subcategory: {
              include: { category: true },
            },
            category: true,
          },
        },
      },
    }),
  ])

  // ── 3. Построить ratesMap ───────────────────────────────────
  const defaultRates = {
    wbWalletPct: 2.0,
    wbAcquiringPct: 2.7,
    wbJemPct: 1.0,
    wbCreditPct: 7.0,
    wbOverheadPct: 6.0,
    wbTaxPct: 8.0,
  }
  const rates = { ...defaultRates }
  for (const setting of appSettings) {
    const parsed = parseFloat(setting.value)
    if (!Number.isNaN(parsed)) {
      ;(rates as Record<string, number>)[setting.key] = parsed
    }
  }

  // ── 4. Собрать linkedNmIds ──────────────────────────────────
  const articleToProduct = new Map<
    number,
    (typeof linkedArticles)[number]["product"]
  >()
  for (const a of linkedArticles) {
    const nmId = parseInt(a.article, 10)
    if (!Number.isNaN(nmId)) {
      // Если несколько MarketplaceArticle ссылаются на один nmId — берём первый
      if (!articleToProduct.has(nmId)) {
        articleToProduct.set(nmId, a.product)
      }
    }
  }
  const linkedNmIds = Array.from(articleToProduct.keys())

  // ── 5. Загрузить WbCards + CalculatedPrice ──────────────────
  const wbCards =
    linkedNmIds.length > 0
      ? await prisma.wbCard.findMany({ where: { nmId: { in: linkedNmIds } } })
      : []

  const wbCardIds = wbCards.map((c) => c.id)
  const calculatedPrices =
    wbCardIds.length > 0
      ? await prisma.calculatedPrice.findMany({
          where: { wbCardId: { in: wbCardIds } },
        })
      : []

  // ── 6. Группировать WbCards по Product ──────────────────────
  // productId → WbCard[]
  const productToCards = new Map<
    string,
    Array<{
      card: (typeof wbCards)[number]
      product: (typeof linkedArticles)[number]["product"]
    }>
  >()
  for (const card of wbCards) {
    const product = articleToProduct.get(card.nmId)
    if (!product) continue
    if (!productToCards.has(product.id)) {
      productToCards.set(product.id, [])
    }
    productToCards.get(product.id)!.push({ card, product })
  }

  // ── 7. Построить ProductGroup[] ─────────────────────────────
  const groups: ProductGroup[] = []

  for (const [productId, cardRefs] of productToCards) {
    const firstProduct = cardRefs[0].product
    const cardGroups: WbCardRowGroup[] = []

    for (const { card, product } of cardRefs) {
      // Resolved per-product параметры (fallback chain)
      const resolvedDrr = resolveDrrPct({
        productOverride: product.drrOverridePct ?? null,
        subcategoryDefault: product.subcategory?.defaultDrrPct ?? null,
      })
      const resolvedDefect = resolveDefectRatePct({
        productOverride: product.defectRateOverridePct ?? null,
        categoryDefault: product.category?.defaultDefectRatePct ?? null,
      })
      const resolvedDelivery = resolveDeliveryCostRub(
        product.deliveryCostRub ?? null,
      )

      // Базовые inputs для всех priceRows одной карты (ставки + per-product)
      const baseInputs: Omit<
        PricingInputs,
        "priceBeforeDiscount" | "sellerDiscountPct"
      > = {
        wbDiscountPct: card.discountWb ?? 0,
        clubDiscountPct: card.clubDiscount ?? 0,
        commFbwPct: card.commFbwIu ?? card.commFbwStd ?? 0,
        costPrice: product.cost?.costPrice ?? 0,
        buyoutPct: card.buyoutPercent ?? 100,
        drrPct: resolvedDrr,
        defectRatePct: resolvedDefect,
        deliveryCostRub: resolvedDelivery,
        walletPct: rates.wbWalletPct,
        acquiringPct: rates.wbAcquiringPct,
        jemPct: rates.wbJemPct,
        creditPct: rates.wbCreditPct,
        overheadPct: rates.wbOverheadPct,
        taxPct: rates.wbTaxPct,
      }

      const priceRows: PriceRow[] = []

      // a) Текущая цена
      const currentSellerPrice = card.priceBeforeDiscount ?? 0
      const currentSellerDiscount = card.sellerDiscount ?? 0
      priceRows.push({
        id: `${card.id}-current`,
        type: "current",
        sellerPriceBeforeDiscount: currentSellerPrice,
        label: "Текущая",
        computed: calculatePricing({
          ...baseInputs,
          priceBeforeDiscount: currentSellerPrice,
          sellerDiscountPct: currentSellerDiscount,
        }),
      })

      // b) Regular акции для этой nmId
      const regularRows: PriceRow[] = []
      for (const promo of promotions) {
        if (promo.type === "auto") continue
        const nom = promo.nomenclatures.find((n) => n.nmId === card.nmId)
        if (!nom || nom.planPrice == null) continue

        // В акции цена это planPrice, скидка = planDiscount
        regularRows.push({
          id: `${card.id}-regular-${promo.id}`,
          type: "regular",
          sellerPriceBeforeDiscount: nom.planPrice,
          label: promo.name,
          promotionDescription: promo.description,
          promotionAdvantages: promo.advantages,
          computed: calculatePricing({
            ...baseInputs,
            priceBeforeDiscount: nom.planPrice,
            sellerDiscountPct: nom.planDiscount ?? 0,
          }),
        })
      }
      // Сортировка DESC по planPrice
      regularRows.sort(
        (a, b) => b.sellerPriceBeforeDiscount - a.sellerPriceBeforeDiscount,
      )
      priceRows.push(...regularRows)

      // c) Auto акции для этой nmId (только те, у которых planPrice задан — из Excel)
      const autoRows: PriceRow[] = []
      for (const promo of promotions) {
        if (promo.type !== "auto") continue
        const nom = promo.nomenclatures.find((n) => n.nmId === card.nmId)
        if (!nom || nom.planPrice == null) continue

        autoRows.push({
          id: `${card.id}-auto-${promo.id}`,
          type: "auto",
          sellerPriceBeforeDiscount: nom.planPrice,
          label: promo.name,
          promotionDescription: promo.description,
          promotionAdvantages: promo.advantages,
          computed: calculatePricing({
            ...baseInputs,
            priceBeforeDiscount: nom.planPrice,
            sellerDiscountPct: nom.planDiscount ?? 0,
          }),
        })
      }
      autoRows.sort(
        (a, b) => b.sellerPriceBeforeDiscount - a.sellerPriceBeforeDiscount,
      )
      priceRows.push(...autoRows)

      // d) Расчётные цены (slot 1, 2, 3)
      const cardCalcs = calculatedPrices
        .filter((cp) => cp.wbCardId === card.id)
        .sort((a, b) => a.slot - b.slot)

      for (const cp of cardCalcs) {
        // Для расчётки используем её собственные overrides (если заданы)
        // иначе — fallback chain с product/subcategory/category
        const cpDrr = cp.drrPct ?? resolvedDrr
        const cpDefect = cp.defectRatePct ?? resolvedDefect
        const cpDelivery = cp.deliveryCostRub ?? resolvedDelivery

        priceRows.push({
          id: `${card.id}-calc-${cp.slot}`,
          type: "calculated",
          sellerPriceBeforeDiscount: cp.sellerPrice,
          label: cp.name,
          calculatedSlot: cp.slot as 1 | 2 | 3,
          computed: calculatePricing({
            ...baseInputs,
            drrPct: cpDrr,
            defectRatePct: cpDefect,
            deliveryCostRub: cpDelivery,
            priceBeforeDiscount: cp.sellerPrice,
            sellerDiscountPct: 0, // расчётная цена уже после скидки продавца
          }),
        })
      }

      cardGroups.push({
        card: {
          id: card.id,
          nmId: card.nmId,
          label: null, // EXECUTOR: если у WbCard есть поле label (ярлык) — взять его. Иначе null.
        },
        priceRows,
      })
    }

    // Агрегаты по Product
    const totalStock = cardRefs.reduce(
      (s, { card }) => s + (card.stockQty ?? 0),
      0,
    )
    const totalAvgSalesSpeed = cardRefs.reduce(
      (s, { card }) => s + (card.avgSalesSpeed7d ?? 0),
      0,
    )
    const totalRowsInProduct = cardGroups.reduce(
      (s, cg) => s + cg.priceRows.length,
      0,
    )

    groups.push({
      product: {
        id: firstProduct.id,
        name: firstProduct.name,
        photoUrl: firstProduct.photoUrl ?? null,
        totalStock,
        totalAvgSalesSpeed,
      },
      cards: cardGroups,
      totalRowsInProduct,
    })
  }

  // Сортировка групп по названию Product (alphabetical для детерминизма)
  groups.sort((a, b) => a.product.name.localeCompare(b.product.name, "ru"))

  return (
    <div className="space-y-4">
      <GlobalRatesBar initialRates={rates} />

      {/* Кнопки шапки. WbPromotionsSyncButton + WbAutoPromoUploadButton добавятся в 07-10 */}
      <div className="flex flex-wrap gap-2">
        <WbSyncButton />
        <WbSyncSppButton />
        {/* TODO (план 07-10): <WbPromotionsSyncButton /> */}
```

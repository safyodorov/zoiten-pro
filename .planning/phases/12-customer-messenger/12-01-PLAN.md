    ```sql
    -- Phase 12: Профиль покупателя + Мессенджеры
    -- enum MessengerType + SupportTicket.messengerType/messengerContact + backfill CHAT тикетов

    -- 1. CreateEnum
    CREATE TYPE "MessengerType" AS ENUM ('TELEGRAM', 'WHATSAPP', 'OTHER');

    -- 2. AlterTable SupportTicket — новые поля
    ALTER TABLE "SupportTicket"
      ADD COLUMN "messengerType" "MessengerType",
      ADD COLUMN "messengerContact" TEXT;

    -- 3. Backfill: для всех CHAT тикетов с customerId=NULL и wbExternalId NOT NULL:
    --    - Создать Customer с wbUserId='chat:'||wbExternalId и name=customerNameSnapshot
    --    - Проставить ticket.customerId
    --    Идемпотентно через ON CONFLICT (wbUserId unique).

    -- 3a. Upsert Customer для каждого подходящего тикета (ON CONFLICT DO UPDATE чтобы вернуть id существующего)
    WITH candidate_chats AS (
      SELECT DISTINCT
        'chat:' || t."wbExternalId" AS wb_user_id,
        t."customerNameSnapshot" AS client_name
      FROM "SupportTicket" t
      WHERE t."channel" = 'CHAT'
        AND t."customerId" IS NULL
        AND t."wbExternalId" IS NOT NULL
    ),
    inserted AS (
      INSERT INTO "Customer" ("id", "wbUserId", "name", "createdAt", "updatedAt")
      SELECT
        gen_random_uuid()::text,
        cc.wb_user_id,
        cc.client_name,
        NOW(),
        NOW()
      FROM candidate_chats cc
      ON CONFLICT ("wbUserId") DO UPDATE SET "name" = COALESCE(EXCLUDED."name", "Customer"."name")
      RETURNING "id", "wbUserId"
    )
    UPDATE "SupportTicket" t
    SET "customerId" = c."id"
    FROM "Customer" c
    WHERE c."wbUserId" = 'chat:' || t."wbExternalId"
      AND t."channel" = 'CHAT'
      AND t."customerId" IS NULL
      AND t."wbExternalId" IS NOT NULL;
    ```

    **ВАЖНО (gen_random_uuid):** Postgres 16 имеет `gen_random_uuid()` в pgcrypto или core (Postgres 13+ — core). Если ошибка — alternative: использовать `encode(gen_random_bytes(16), 'hex')` либо генерировать id на уровне приложения (в этом плане не нужно, т.к. миграция применяется один раз и расхождение с cuid-паттерном Prisma не критично — Prisma принимает любые TEXT id).

    ```typescript
    // lib/customer-aggregations.ts — pure helpers для профиля покупателя (Plan 12-02 использует)
    // Без зависимостей, чистые функции — легко тестируются.

    import type { TicketChannel } from "@prisma/client"

    export interface TicketForAggregation {
      channel: TicketChannel
      rating: number | null
    }

    export function countTicketsByChannel(
      tickets: TicketForAggregation[]
    ): Record<TicketChannel, number> {
      const acc = {
        FEEDBACK: 0,
        QUESTION: 0,
        CHAT: 0,
        RETURN: 0,
        MESSENGER: 0,
      } as Record<TicketChannel, number>
      for (const t of tickets) acc[t.channel] = (acc[t.channel] ?? 0) + 1
      return acc
    }

    export function averageFeedbackRating(
      tickets: TicketForAggregation[]
    ): number | null {
      const ratings = tickets
        .filter((t) => t.channel === "FEEDBACK" && t.rating !== null)
        .map((t) => t.rating as number)
      if (ratings.length === 0) return null
      const sum = ratings.reduce((a, b) => a + b, 0)
      return Math.round((sum / ratings.length) * 100) / 100 // 2 decimals
    }
    ```

    ```typescript
    /**
     * Pure helpers для Phase 20 — расчёт сумм и дат платежей в закупках.
     * Используется server actions (createPurchase, addPurchasePayment) и UI
     * (моментальный пересчёт в форме редактирования).
     */

    export type PaymentStatusValue = "PLANNED" | "PAID" | "OVERDUE"

    /**
     * Расчёт суммы платежа от total и процента.
     * Округление: floor до 2 знаков после запятой (Decimal(14, 2)).
     */
    export function computePaymentAmount(percent: number, total: number): number {
      if (percent <= 0 || total <= 0) return 0
      const raw = (percent / 100) * total
      // Round down к копейкам — избегаем over-charge
      return Math.floor(raw * 100) / 100
    }

    /**
     * Дата депозита: createdAt + 3 calendar days (D-08, user prompt verbatim).
     */
    export function computeDepositDueDate(purchaseCreatedAt: Date): Date {
      const d = new Date(purchaseCreatedAt)
      d.setDate(d.getDate() + 3)
      return d
    }

    /**
     * Дата баланса: dueDate депозита + leadTimeDays из SupplierProductLink.
     * Если leadTimeDays null/undefined → возвращаем null (UI отобразит «—» + warning).
     */
    export function computeBalanceDueDate(
      depositDueDate: Date,
      leadTimeDays: number | null | undefined
    ): Date | null {
      if (leadTimeDays == null || leadTimeDays < 0) return null
      const d = new Date(depositDueDate)
      d.setDate(d.getDate() + leadTimeDays)
      return d
    }

    /**
     * Перерасчёт статуса платежа на основе dueDate / paidDate / now.
     * Используется при чтении (server action + UI render) — не персистируется
     * автоматически. Server actions при mutation вызывают recompute и пишут.
     */
    export function recomputeOverdueStatus(payment: {
      dueDate: Date
      paidDate: Date | null
      status: PaymentStatusValue
    }, now: Date = new Date()): PaymentStatusValue {
      if (payment.paidDate != null) return "PAID"
      // Compare on day granularity (МСК)
      const due = new Date(payment.dueDate)
      due.setHours(0, 0, 0, 0)
      const today = new Date(now)
      today.setHours(0, 0, 0, 0)
      if (due.getTime() < today.getTime()) return "OVERDUE"
      return "PLANNED"
    }
    ```

    ```typescript
    export async function createPurchase(data: z.infer<typeof CreatePurchaseSchema>): Promise<CreateResult> {
      try {
        await requireSection("PROCUREMENT", "MANAGE")
        const parsed = CreatePurchaseSchema.parse(data)

        // 1. Items total в указанной currency
        const itemsTotal = parsed.items.reduce((sum, it) => sum + it.quantity * it.unitPrice, 0)

        // 2. Найти SupplierProductLink для default percent + leadTimeDays.
        // Логика: для первого товара (или агрегата по всем) берём depositPct/balancePct/leadTime
        // из productLinks. Если несколько товаров с разными параметрами — берём max leadTime
        // и max depositPct (consversative). Override из parsed.overrides перекрывает всё.
        const productIds = parsed.items.map((i) => i.productId)
        const links = await prisma.supplierProductLink.findMany({
          where: { supplierId: parsed.supplierId, productId: { in: productIds } },
        })

        const defaultDepositPct = parsed.overrides?.depositPct
          ?? (links.length ? Math.max(...links.filter(l => l.depositPct != null).map(l => Number(l.depositPct))) : 30)
        const defaultBalancePct = parsed.overrides?.balancePct
          ?? (links.length ? Math.max(...links.filter(l => l.balancePct != null).map(l => Number(l.balancePct))) : 70)
        const maxLeadTime = links.length
          ? Math.max(...links.filter(l => l.leadTimeDays != null).map(l => l.leadTimeDays as number), 0)
          : 0
        const safeLeadTime = maxLeadTime > 0 ? maxLeadTime : null

        // 3. Транзакция: create Purchase + items + 2 auto-payments
        const purchase = await prisma.$transaction(async (tx) => {
          const now = new Date()
          const created = await tx.purchase.create({
            data: {
              status: "PLANNED",
              currency: parsed.currency,
              supplierId: parsed.supplierId,
              optionsDescription: parsed.optionsDescription?.trim() || null,
              optionsExtraCost: parsed.optionsExtraCost != null ? (parsed.optionsExtraCost as never) : null,
              logisticsCost: parsed.logisticsCost != null ? (parsed.logisticsCost as never) : null,
              logisticsComment: parsed.logisticsComment?.trim() || null,
              items: {
                create: parsed.items.map((it) => ({
                  productId: it.productId,
                  quantity: it.quantity,
                  unitPrice: it.unitPrice as never,
                })),
              },
            },
          })

          // Auto-payments per D-08
          const depositDate = parsed.overrides?.depositDueDate
            ? new Date(parsed.overrides.depositDueDate)
            : computeDepositDueDate(now)
          const balanceDate = parsed.overrides?.balanceDueDate
            ? new Date(parsed.overrides.balanceDueDate)
            : computeBalanceDueDate(depositDate, safeLeadTime)

          // Депозит создаётся всегда
          await tx.purchasePayment.create({
            data: {
              purchaseId: created.id,
              type: "DEPOSIT",
              ordinal: 1,
              percent: defaultDepositPct as never,
              amount: computePaymentAmount(defaultDepositPct, itemsTotal) as never,
              currency: parsed.currency,
              dueDate: depositDate,
              status: "PLANNED",
            },
          })

          // Баланс создаётся если balanceDate известна (leadTime есть) ИЛИ юзер дал override
          if (balanceDate) {
            await tx.purchasePayment.create({
              data: {
                purchaseId: created.id,
                type: "BALANCE",
                ordinal: 1,
                percent: defaultBalancePct as never,
                amount: computePaymentAmount(defaultBalancePct, itemsTotal) as never,
                currency: parsed.currency,
                dueDate: balanceDate,
                status: "PLANNED",
              },
            })
          }

          return created
        })

        revalidatePath("/procurement/purchases")
        revalidatePath(`/procurement/purchases/${purchase.id}`)
        return { ok: true, id: purchase.id }
      } catch (e) {
        const a = handleAuthError(e); if (a) return a
        if (e instanceof z.ZodError) return { ok: false, error: e.issues[0]?.message ?? "Validation" }
        console.error("createPurchase:", e)
        return { ok: false, error: "Ошибка сервера" }
      }
    }
    ```

    ```typescript
    export async function markPaymentPaid(id: string, paidDateStr?: string): Promise<ActionResult> {

    export async function markPaymentUnpaid(id: string): Promise<ActionResult> {
      // Пересчитать статус: если dueDate < now → OVERDUE, иначе PLANNED
      const payment = await prisma.purchasePayment.findUnique({ where: { id } })
      if (!payment) return { ok: false, error: "Платёж не найден" }
      const newStatus = recomputeOverdueStatus({
        dueDate: payment.dueDate,
        paidDate: null,
        status: payment.status as "PLANNED" | "PAID" | "OVERDUE",
      })
      await prisma.purchasePayment.update({
        where: { id },
        data: { paidDate: null, status: newStatus },
      })
      return { ok: true }
    }

    /**
     * Помощник: пересчитать OVERDUE статусы для всех PLANNED платежей закупки.
     * Вызывается при чтении детальной страницы Purchase (server side, перед render).
     */
    export async function recomputeOverduePaymentsForPurchase(purchaseId: string): Promise<void> {
      // Без RBAC — internal helper, вызывается из RSC pages (которые сами имеют requireSection)
      const payments = await prisma.purchasePayment.findMany({
        where: { purchaseId, status: { in: ["PLANNED", "OVERDUE"] }, paidDate: null },
      })
      const now = new Date()
      const updates = payments
        .map((p) => {
          const fresh = recomputeOverdueStatus({
            dueDate: p.dueDate,
            paidDate: null,
            status: p.status as "PLANNED" | "PAID" | "OVERDUE",
          }, now)
          return fresh !== p.status ? { id: p.id, status: fresh } : null
        })
        .filter(Boolean) as Array<{ id: string; status: "PLANNED" | "OVERDUE" }>
      for (const u of updates) {
        await prisma.purchasePayment.update({ where: { id: u.id }, data: { status: u.status } })
      }
    }
    ```

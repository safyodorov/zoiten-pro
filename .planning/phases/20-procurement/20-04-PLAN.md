    ```tsx
    import { requireSection } from "@/lib/rbac"
    import { prisma } from "@/lib/prisma"
    import { SuppliersFilters } from "@/components/procurement/SuppliersFilters"
    import { SuppliersTable } from "@/components/procurement/SuppliersTable"
    import { CreateSupplierButton } from "@/components/procurement/CreateSupplierButton"

    type SearchParams = { buyerIds?: string; directionIds?: string; brandIds?: string; categoryIds?: string; subcategoryIds?: string }

    export default async function SuppliersListPage({ searchParams }: { searchParams: Promise<SearchParams> }) {
      await requireSection("PROCUREMENT")
      const sp = await searchParams

      const buyerIds = sp.buyerIds?.split(",").filter(Boolean) ?? []
      const directionIds = sp.directionIds?.split(",").filter(Boolean) ?? []
      const brandIds = sp.brandIds?.split(",").filter(Boolean) ?? []
      const categoryIds = sp.categoryIds?.split(",").filter(Boolean) ?? []
      const subcategoryIds = sp.subcategoryIds?.split(",").filter(Boolean) ?? []

      // Каскадная фильтрация через productLinks → product → brand/category/...
      const where: any = { deletedAt: null }
      if (buyerIds.length) where.buyerEmployeeId = { in: buyerIds }
      if (directionIds.length || brandIds.length || categoryIds.length || subcategoryIds.length) {
        where.productLinks = { some: { product: {
          ...(brandIds.length ? { brandId: { in: brandIds } } : {}),
          ...(directionIds.length ? { brand: { directionId: { in: directionIds } } } : {}),
          ...(categoryIds.length ? { categoryId: { in: categoryIds } } : {}),
          ...(subcategoryIds.length ? { subcategoryId: { in: subcategoryIds } } : {}),
        }}}
      }

      const [suppliers, buyers, directions, brands, categories, subcategories] = await Promise.all([
        prisma.supplier.findMany({
          where,
          include: {
            buyerEmployee: { select: { id: true, lastName: true, firstName: true } },
            contacts: { where: { isPrimary: true } },
            productLinks: { select: { id: true } },
            negotiations: { select: { id: true } },
          },
          orderBy: [
            { buyerEmployee: { lastName: "asc" } },
            { nameForeign: "asc" },
          ],
        }),
        prisma.employee.findMany({ where: { fireDate: null }, select: { id: true, lastName: true, firstName: true }, orderBy: [{ lastName: "asc" }, { firstName: "asc" }] }),
        prisma.productDirection.findMany({ select: { id: true, name: true }, orderBy: { sortOrder: "asc" } }),
        prisma.brand.findMany({ select: { id: true, name: true, directionId: true }, orderBy: { sortOrder: "asc" } }),
        prisma.category.findMany({ select: { id: true, name: true, brandId: true }, orderBy: { sortOrder: "asc" } }),
        prisma.subcategory.findMany({ select: { id: true, name: true, categoryId: true }, orderBy: { sortOrder: "asc" } }),
      ])

      return (
        <div className="h-full flex flex-col">
          <div className="flex items-center gap-3 px-4 py-2 border-b">
            <SuppliersFilters
              buyers={buyers}
              directions={directions}
              brands={brands}
              categories={categories}
              subcategories={subcategories}
              selected={{ buyerIds, directionIds, brandIds, categoryIds, subcategoryIds }}
            />
            <CreateSupplierButton buyers={buyers} />
          </div>
          <div className="flex-1 min-h-0">
            <SuppliersTable suppliers={suppliers} />
          </div>
        </div>
      )
    }
    ```

    **A) `app/(dashboard)/procurement/suppliers/[id]/page.tsx`** (RSC) — детальная страница с loads всех связанных данных:

    ```tsx
    import { requireSection } from "@/lib/rbac"
    import { prisma } from "@/lib/prisma"
    import { notFound } from "next/navigation"
    import { SupplierDetailTabs } from "@/components/procurement/SupplierDetailTabs"

    export default async function SupplierDetailPage({ params }: { params: Promise<{id: string}> }) {
      await requireSection("PROCUREMENT")
      const { id } = await params

      const [supplier, buyers, allProducts] = await Promise.all([
        prisma.supplier.findUnique({
          where: { id },
          include: {
            buyerEmployee: { select: { id: true, lastName: true, firstName: true } },
            contacts: { orderBy: [{ type: "asc" }, { isPrimary: "desc" }, { name: "asc" }] },
            productLinks: { include: { product: { select: { id: true, name: true, article: true, sku: true } } } },
            negotiations: {
              include: {
                products: { include: { product: { select: { id: true, name: true, article: true } } } },
                participants: {
                  include: {
                    employee: { select: { lastName: true, firstName: true } },
                    supplierContact: { select: { name: true, type: true } },
                  },
                },
              },
              orderBy: { date: "desc" },
            },
          },
        }),
        prisma.employee.findMany({ where: { fireDate: null }, select: { id: true, lastName: true, firstName: true }, orderBy: [{ lastName: "asc" }, { firstName: "asc" }] }),
        prisma.product.findMany({
          where: { deletedAt: null },
          select: { id: true, name: true, article: true, sku: true },
          orderBy: [{ sku: "asc" }],
        }),
      ])

      if (!supplier || supplier.deletedAt) notFound()

      return (
        <div className="h-full overflow-auto">
          <SupplierDetailTabs supplier={supplier} buyers={buyers} allProducts={allProducts} />
        </div>
      )
    }
    ```

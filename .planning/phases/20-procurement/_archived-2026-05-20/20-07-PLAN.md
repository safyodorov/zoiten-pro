      return (

            const active = pathname === t.href || pathname.startsWith(`${t.href}/`)
            return (
              <Link
                key={t.href}
                href={t.href}
                prefetch={false}
                className={cn(
                  "px-3 py-2 text-sm border-b-2 transition-colors",
                  active
                    ? "border-primary text-foreground"
                    : "border-transparent text-muted-foreground hover:text-foreground hover:border-muted-foreground/30"
                )}
              >
                {t.label}
              </Link>
            )
          })}
        </nav>
      )
    }

    A) app/(dashboard)/procurement/plan/page.tsx (RSC):

    ```tsx
    import { requireSection } from "@/lib/rbac"
    import { prisma } from "@/lib/prisma"
    import { getLatestRate } from "@/lib/cbr-rates"
    import { ProcurementTabs } from "@/components/procurement/ProcurementTabs"
    import { PlanSummaryCards } from "@/components/procurement/PlanSummaryCards"
    import { PlanForecastTable } from "@/components/procurement/PlanForecastTable"
    import { recomputeOverdueStatus } from "@/lib/purchase-payments"

    export default async function PlanPage() {
      await requireSection("PROCUREMENT")

      // Загружаем PLANNED закупки + их payments
      const purchases = await prisma.purchase.findMany({
        where: { status: { in: ["PLANNED", "ACTIVE"] } },
        include: {
          supplier: { select: { id: true, nameEnglish: true, nameForeign: true } },
          items: { select: { quantity: true, unitPrice: true } },
          payments: { where: { paidDate: null }, orderBy: { dueDate: "asc" } },
        },
      })

      // Собираем все unique currencies из payments
      const currencies = new Set<string>()
      for (const p of purchases) for (const pay of p.payments) currencies.add(pay.currency)
      const rates = new Map<string, number>()
      for (const code of currencies) {
        const r = await getLatestRate(code)
        if (r) rates.set(code, Number(r.rateToRub) / r.nominal)
        else rates.set(code, code === "RUB" ? 1 : 0)
      }

      const toRub = (amount: number, currency: string) => amount * (rates.get(currency) ?? 0)

      // Aggregation per month (key = YYYY-MM)
      const now = new Date()
      const months = new Map<string, { paymentsCount: number; totalRub: number; suppliers: Map<string, number> }>()
      let totalPlannedRub = 0
      let nearestDeposit: { dueDate: Date; supplierName: string; amount: number; currency: string } | null = null
      let overdueCount = 0
      let overdueAmountRub = 0

      for (const purchase of purchases) {
        for (const pay of purchase.payments) {
          const amountNum = Number(pay.amount)
          const rub = toRub(amountNum, pay.currency)
          totalPlannedRub += rub

          // OVERDUE detection (use helper)
          const status = recomputeOverdueStatus({
            dueDate: pay.dueDate,
            paidDate: null,
            status: pay.status as "PLANNED" | "PAID" | "OVERDUE",
          }, now)
          if (status === "OVERDUE") {
            overdueCount += 1
            overdueAmountRub += rub
          }

          if (!nearestDeposit && pay.type === "DEPOSIT" && status !== "OVERDUE") {
            nearestDeposit = {
              dueDate: pay.dueDate,
              supplierName: purchase.supplier.nameEnglish,
              amount: amountNum,
              currency: pay.currency,
            }
          }

          const monthKey = `${pay.dueDate.getUTCFullYear()}-${String(pay.dueDate.getUTCMonth() + 1).padStart(2, "0")}`
          const m = months.get(monthKey) ?? { paymentsCount: 0, totalRub: 0, suppliers: new Map() }
          m.paymentsCount += 1
          m.totalRub += rub
          m.suppliers.set(purchase.supplier.nameEnglish, (m.suppliers.get(purchase.supplier.nameEnglish) ?? 0) + rub)
          months.set(monthKey, m)
        }
      }

      // Top supplier по totalRub
      const supplierTotals = new Map<string, { id: string; nameEnglish: string; amount: number }>()
      for (const p of purchases) {
        const id = p.supplier.id
        const ex = supplierTotals.get(id) ?? { id, nameEnglish: p.supplier.nameEnglish, amount: 0 }
        for (const pay of p.payments) ex.amount += toRub(Number(pay.amount), pay.currency)
        supplierTotals.set(id, ex)
      }
      const topSupplier = Array.from(supplierTotals.values()).sort((a, b) => b.amount - a.amount)[0] ?? null

      // Превратить months Map в sorted array (DESC)
      const monthRows = Array.from(months.entries())
        .sort(([a], [b]) => a.localeCompare(b))
        .map(([monthKey, data]) => ({
          monthKey,
          paymentsCount: data.paymentsCount,
          totalRub: data.totalRub,
          topSuppliers: Array.from(data.suppliers.entries())
            .sort((a, b) => b[1] - a[1])
            .slice(0, 3)
            .map(([name, amount]) => ({ name, amount })),
        }))

      return (
        <div className="h-full flex flex-col">
          <ProcurementTabs />
          <div className="px-4 py-3 space-y-3">
            <PlanSummaryCards
              totalPlannedRub={totalPlannedRub}
              topSupplier={topSupplier}
              nearestDeposit={nearestDeposit}
              overdueCount={overdueCount}
              overdueAmountRub={overdueAmountRub}
            />
          </div>
          <div className="flex-1 min-h-0 px-4 pb-4">
            <PlanForecastTable rows={monthRows} />
          </div>
        </div>
      )
    }
    ```

    export async function createSupplier(data: z.infer<typeof SupplierSchema>): Promise<CreateResult> {
      try {
        await requireSection("PROCUREMENT", "MANAGE")
        const parsed = SupplierSchema.parse(data)
        const created = await prisma.supplier.create({
          data: {
            nameForeign: parsed.nameForeign.trim(),
            nameEnglish: parsed.nameEnglish.trim(),
            buyerEmployeeId: parsed.buyerEmployeeId ?? null,
            cooperationSummary: parsed.cooperationSummary?.trim() || null,
          },

        revalidatePath("/procurement/suppliers")
        return { ok: true, id: created.id }

      }
    }

    export async function createSupplierContact(data: z.infer<typeof SupplierContactBaseSchema>): Promise<CreateResult> {
      try {
        await requireSection("PROCUREMENT", "MANAGE")
        const parsed = SupplierContactBaseSchema.parse(data)
        // Если isPrimary=true — в транзакции снять isPrimary с других того же type+supplier
        const contact = await prisma.$transaction(async (tx) => {
          if (parsed.isPrimary) {
            await tx.supplierContact.updateMany({
              where: { supplierId: parsed.supplierId, type: parsed.type },
              data: { isPrimary: false },
            })
          }
          return tx.supplierContact.create({
            data: {
              supplierId: parsed.supplierId,
              type: parsed.type,
              name: parsed.name.trim(),
              phone: parsed.phone?.trim() || null,
              preferredContact: parsed.preferredContact,
              preferredContactCustom: parsed.preferredContact === "OTHER"
                ? parsed.preferredContactCustom!.trim()
                : null,
              description: parsed.description?.trim() || null,
              isPrimary: parsed.isPrimary,
            },
          })
        })
        revalidatePath(`/procurement/suppliers/${parsed.supplierId}`)
        return { ok: true, id: contact.id }
      } catch (e) {
        const a = handleAuthError(e); if (a) return a
        if (e instanceof z.ZodError) return { ok: false, error: e.issues[0]?.message ?? "Validation" }
        console.error("createSupplierContact:", e)
        return { ok: false, error: "Ошибка сервера" }
      }
    }

    export async function setPrimarySupplierContact(contactId: string): Promise<ActionResult> {
      try {
        await requireSection("PROCUREMENT", "MANAGE")
        await prisma.$transaction(async (tx) => {
          const contact = await tx.supplierContact.findUnique({ where: { id: contactId } })
          if (!contact) throw Object.assign(new Error("Not found"), { code: "P2025" })
          // Снять isPrimary с других contacts того же supplierId+type
          await tx.supplierContact.updateMany({
            where: {
              supplierId: contact.supplierId,
              type: contact.type,
              id: { not: contactId },
            },
            data: { isPrimary: false },
          })
          await tx.supplierContact.update({
            where: { id: contactId },
            data: { isPrimary: true },
          })
        })
        // revalidate detail page
        return { ok: true }
      } catch (e) {
        const a = handleAuthError(e); if (a) return a
        if ((e as { code?: string })?.code === "P2025") return { ok: false, error: "Контакт не найден" }
        console.error("setPrimarySupplierContact:", e)
        return { ok: false, error: "Ошибка сервера" }
      }
    }

    ```typescript
    export default async function ProductsPage({
      searchParams,
    }: {
      searchParams: Promise<{ q?: string; page?: string; status?: string }>
    }) {
      await requireSection("PRODUCTS")
      const { q, page: pageParam, status } = await searchParams
    ```

    Build where clause:
    - Default (no status param or status="IN_STOCK"): { deletedAt: null, availability: "IN_STOCK" }
    - status="OUT_OF_STOCK": { deletedAt: null, availability: "OUT_OF_STOCK" }
    - status="DISCONTINUED": { deletedAt: null, availability: "DISCONTINUED" }
    - status="DELETED": { deletedAt: { not: null } }
    - status="ALL": { deletedAt: null } (excludes deleted — "All active")
    - Add name filter if q: add `name: { contains: q, mode: "insensitive" }` to where

    Fetch:
    ```typescript
    const PAGE_SIZE = 20
    const currentPage = Math.max(1, parseInt(pageParam ?? "1", 10))
    const skip = (currentPage - 1) * PAGE_SIZE

    const [products, total] = await Promise.all([
      prisma.product.findMany({
        where,
        include: { brand: true, category: true },
        orderBy: { createdAt: "desc" },
        skip,
        take: PAGE_SIZE,
      }),
      prisma.product.count({ where }),
    ])
    const totalPages = Math.ceil(total / PAGE_SIZE)
    ```

    Render:
    ```tsx
    return (
      <div className="space-y-4">
        <div className="flex items-center justify-between">
          <h1 className="text-2xl font-semibold">Товары</h1>
          <Link href="/products/new" className="...button styles...">+ Добавить товар</Link>
        </div>
        <ProductStatusTabs currentStatus={status ?? "IN_STOCK"} />
        <ProductSearchInput defaultValue={q ?? ""} />
        <ProductsTable
          products={products}
          currentPage={currentPage}
          totalPages={totalPages}
          currentStatus={status ?? "IN_STOCK"}
          searchQuery={q ?? ""}
        />
      </div>
    )
    ```

    Use shadcn Button for the "Добавить товар" link (import from @/components/ui/button) — but render as Link since Button lacks asChild (per STATE.md decision). Style the Link directly with className using Tailwind button classes: "inline-flex items-center gap-2 rounded-md bg-primary text-primary-foreground px-4 py-2 text-sm font-medium hover:bg-primary/90".
